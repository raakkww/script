-- AI PREDICTOR
-- âââââââââââââââââââââââââââââââââââââââ
local function scoreReceiver(p)
if not p.Character then return 0,nil end
local r=p.Character:FindFirstChild("HumanoidRootPart")
local h=p.Character:FindFirstChildOfClass("Humanoid")
if not r or not h or h.Health<=0 then return 0,nil end
local s=0
local minDef=math.huge
for _,d in ipairs(playerCache) do
if d~=LP and d.Team~=LP.Team and d.Character then
local dr=d.Character:FindFirstChild("HumanoidRootPart")
if dr then local dd=(dr.Position-r.Position).Magnitude if dd<minDef then minDef=dd end end
end
end
s+=math.min(minDef,30)2
s+=math.max(0,50-math.abs(200-math.abs(r.Position.Z)))1.5
local vel=r.AssemblyLinearVelocity
local dot=vel:Dot((Vector3.new(0,0,200)-r.Position).Unit)
if dot>0 then s+=dot0.5 end
local dist=(r.Position-Root.Position).Magnitude
if dist>10 and dist<80 then s+=15
elseif dist<=10 then s-=20
elseif dist>80 then s-=10 end
local td=(r.Position-Root.Position).Unit
for _,d in ipairs(playerCache) do
if d~=LP and d.Team~=LP.Team and d.Character then
local dr=d.Character:FindFirstChild("HumanoidRootPart")
if dr then
local tod=dr.Position-Root.Position
local proj=tod:Dot(td)
if proj>0 and proj<dist and (tod-tdproj).Magnitude<4 then s-=25 break end
end
end
end
return s, r.Position+vel*(dist/60)
end

if p~=LP and p.Team==LP.Team then

-- Throw arc + indicator
local function makeIndicator()
if throwIndicator then return end
throwIndicator=Instance.new("Part")
throwIndicator.Size=Vector3.new(1,1,1) throwIndicator.Shape=Enum.PartType.Ball
throwIndicator.Anchored=true throwIndicator.CanCollide=false
throwIndicator.Material=Enum.Material.Neon throwIndicator.CastShadow=false
throwIndicator.Transparency=1 throwIndicator.Parent=Workspace
local pl=Instance.new("PointLight",throwIndicator) pl.Range=20 pl.Brightness=5
end

local function clearArc()
for _,p in ipairs(trailParts) do pcall(function() p:Destroy() end) end trailParts={}
end

local function drawArc(from,to,score)
clearArc()
local norm=math.clamp(score/100,0,1)
for i=0,12 do
local t=i/12
local pos=from:Lerp(to,tt(3-2t))+Vector3.new(0,math.sin(tmath.pi)*10,0)
local dot=Instance.new("Part")
dot.Size=Vector3.new(0.4,0.4,0.4) dot.Shape=Enum.PartType.Ball
dot.Anchored=true dot.CanCollide=false dot.CastShadow=false dot.Material=Enum.Material.Neon
dot.Color=Color3.fromRGB(math.floor(lerp(255,0,norm)),math.floor(lerp(0,255,norm)),0)
dot.Transparency=lerp(0.1,0.85,t) dot.Position=pos dot.Parent=Workspace
table.insert(trailParts,dot)
end
end

-- RenderStepped â aimbot + auto pass state machine
RunService.RenderStepped:Connect(function()
local showArc = CFG.QBAimbot or CFG.AutoPass

if not showArc then  
    if throwIndicator then throwIndicator.Transparency=1 end  
    clearArc() passState="IDLE" return  
end  
if not throwIndicator then makeIndicator() end  

if bestTarget and bestTarget.Character and throwVec~=Vector3.zero then  
    throwIndicator.Position=throwVec+Vector3.new(0,4,0)  
    throwIndicator.Transparency=0.1  
    local norm=math.clamp(bestScore/100,0,1)  
    throwIndicator.Color=Color3.fromRGB(math.floor(lerp(255,0,norm)),math.floor(lerp(0,255,norm)),0)  
    throwIndicator.PointLight.Color=throwIndicator.Color  
    drawArc(Root.Position+Vector3.new(0,5,0),throwVec,bestScore)  

    -- Gentle nudge only (when aimbot on, autopass off)  
    if CFG.CameraNudge and not CFG.AutoPass then  
        local dir=(throwVec-Camera.CFrame.Position).Unit  
        Camera.CFrame=CFrame.new(Camera.CFrame.Position,Camera.CFrame.Position+Camera.CFrame.LookVector:Lerp(dir,0.05))  
    end  
else  
    throwIndicator.Transparency=1 clearArc()  
end  

-- AUTO PASS STATE MACHINE  
if CFG.AutoPass and Root and Hum then  
    local hasBall=ball and (ball.Position-Root.Position).Magnitude<6  
    if not hasBall then passState="IDLE" return end  
    if not bestTarget or throwVec==Vector3.zero then return end  
    if CFG.PassOnlyOpen and bestScore<CFG.PassScoreThreshold then return end  
    if tick()-lastPassTime<1.5 then return end  

    if passState=="IDLE" then  
        passState="AIMING"  
        passCooldown=tick()+CFG.PassDelay  

    elseif passState=="AIMING" then  
        -- Hard lock camera on predicted catch position  
        local aimPos=throwVec+Vector3.new(0,3,0)  
        local dir=(aimPos-Camera.CFrame.Position).Unit  
        Camera.CFrame=CFrame.new(Camera.CFrame.Position,Camera.CFrame.Position+dir)  
        if tick()>=passCooldown then passState="THROWING" end  

    elseif passState=="THROWING" then  
        passState="IDLE" lastPassTime=tick()  
        task.spawn(function()  
            task.wait(0.05)  
            simulateThrow()  
        end)  
    end  
end

end)

local function startFly()
if flyActive then return end flyActive=true
pcall(function()
bv=Instance.new("BodyVelocity",Root)
bv.MaxForce=Vector3.new(1e9,1e9,1e9) bv.Velocity=Vector3.zero
bg=Instance.new("BodyGyro",Root)
bg.MaxTorque=Vector3.new(1e9,1e9,1e9) bg.P=1e6 bg.CFrame=Root.CFrame
end)
task.spawn(function()
while flyActive do
local d=Vector3.zero
if UserInputService:IsKeyDown(Enum.KeyCode.W) then d+=Camera.CFrame.LookVector end
if UserInputService:IsKeyDown(Enum.KeyCode.S) then d-=Camera.CFrame.LookVector end
if UserInputService:IsKeyDown(Enum.KeyCode.A) then d-=Camera.CFrame.RightVector end
if UserInputService:IsKeyDown(Enum.KeyCode.D) then d+=Camera.CFrame.RightVector end
if UserInputService:IsKeyDown(Enum.KeyCode.Space) then d+=Vector3.new(0,1,0) end
if UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then d-=Vector3.new(0,1,0) end
if bv then bv.Velocity=d.Magnitude>0 and d.Unit*CFG.FlySpeed or Vector3.zero end
if bg then bg.CFrame=CFrame.new(Root.Position,Root.Position+Camera.CFrame.LookVector) end
task.wait()
end
end)
end
local function stopFly()
flyActive=false
if bv then bv:Destroy() bv=nil end
if bg then bg:Destroy() bg=nil end
end
