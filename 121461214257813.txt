-- DAMAGE HELPER FUNCTION (FIXED - ReplicateShot + BulletHit every time)
local function doDamage(targetHum)
	if not targetHum or not targetHum.Parent or targetHum.Health <= 0 then return end
	
	-- AUTO FIRE SHOT ANYWHERE (ReplicateShot) - fixes all glitches & enables weaponless
	pcall(function()
		local dummyRep = Instance.new("Tool")
		local shotPos = Vector3.new(75.04045104980469, 4.21449089050293, -405.43402099609375)
		local root = Plr.Character and Plr.Character:FindFirstChild("HumanoidRootPart")
		if root then shotPos = root.Position end
		RS:WaitForChild("ReplicateShot"):FireServer(dummyRep, shotPos)
	end)
	
	-- BulletHit with real tool or dummy (weaponless)
	local hitTool = nil
	if variables.weaponless then
		hitTool = Instance.new("Tool")
	else
		-- try equipped
		local char = Plr.Character
		if char then
			hitTool = char:FindFirstChild(variables.currentWeapon)
		end
		-- fallback backpack + equip
		if not hitTool then
			local bp = Plr.Backpack
			if bp then
				hitTool = bp:FindFirstChild(variables.currentWeapon)
				if hitTool and char and char:FindFirstChild("Humanoid") then
					char.Humanoid:EquipTool(hitTool)
					task.wait(0.05)
				end
			end
		end
	end
	
	if hitTool then
		pcall(function()
			RS:WaitForChild("BulletHit"):FireServer(targetHum, hitTool, false)
		end)
	end
end

local function RefreshPlayerList()
	for _, btn in ipairs(playerButtons) do btn:Destroy() end
	playerButtons = {}
	if not scpTeam then return end
	for _, p in ipairs(Players:GetPlayers()) do
		if p.Team == scpTeam and p.Character and p.Character:FindFirstChild("Humanoid") then
			local btn = Instance.new("TextButton")
			btn.Size = UDim2.new(1, -32, 0, 50)
			btn.BackgroundColor3 = Color3.fromRGB(65, 25, 25)
			btn.Text = "👤 " .. p.Name .. "  |  HP: " .. math.floor(p.Character.Humanoid.Health)
			btn.TextColor3 = Color3.fromRGB(255, 100, 100)
			btn.Font = Enum.Font.Gotham
			btn.TextSize = 17
			btn.TextXAlignment = Enum.TextXAlignment.Left
			btn.Parent = OPPage
			Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 14)
			local s = Instance.new("UIStroke", btn)
			s.Color = Color3.fromRGB(255, 80, 80)
			s.Thickness = 1.8
			
			btn.MouseButton1Click:Connect(function()
				variables.selectedPlayer = p
				SelectedLabel.Text = "Selected: " .. p.Name .. " (HP: " .. math.floor(p.Character.Humanoid.Health) .. ")"
			end)
			table.insert(playerButtons, btn)
		end
	end
	if #playerButtons == 0 then
		local none = Instance.new("TextLabel")
		none.Size = UDim2.new(1, -32, 0, 48)
		none.BackgroundTransparency = 1
		none.Text = "No SCP players online"
		none.TextColor3 = Color3.fromRGB(110, 110, 130)
		none.Font = Enum.Font.Gotham
		none.TextSize = 16
		none.Parent = OPPage

	end
end
