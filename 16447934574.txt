local RunService = game:GetService("RunService")
local Workspace = game:GetService("Workspace")

local Ball = Workspace.FootballField:FindFirstChild("SoccerBall")
local TrailFolder = Workspace:FindFirstChild("BallTrail") or Instance.new("Folder", Workspace)
TrailFolder.Name = "BallTrail"


local TRAIL_SEGMENTS = 50
local TIME_STEP = 0.1
local TRAIL_PART_SIZE = Vector3.new(2,2,2)
local TRAIL_COLOR = Color3.fromRGB(255,0,0)
local MAX_BOUNCES = 3


local FLOOR_Y = 5.473

local function ballBouncedFloor(pos)
    return pos.Y <= FLOOR_Y
end

local function ballHitsGoalFrame(pos, vel)
    return false, pos, vel
end

local function soccerBallTrajectory(dt, vel, pos, gravity, rotation)
    local nextPos = pos + vel * dt + Vector3.new(0, -0.5 * gravity * dt^2, 0)
    local nextVel = Vector3.new(vel.X, vel.Y - gravity * dt, vel.Z)
    return nextPos, nextVel
end
-- --- End of copied functions ---

-- Simulate trajectory with bounces
local function simulateTrajectoryWithBounces(position, velocity)
    local points = {}
    local pos = position
    local vel = velocity
    local bounces = 0

    for i = 1, TRAIL_SEGMENTS do
        local nextPos, nextVel = soccerBallTrajectory(TIME_STEP, vel, pos, 9.81, 0)
        local hit, newPos, newVel = ballHitsGoalFrame(nextPos, nextVel)
        if hit then
            nextPos = newPos
            nextVel = newVel
        end

        if ballBouncedFloor(nextPos) and bounces < MAX_BOUNCES then
            nextVel = Vector3.new(nextVel.X, -nextVel.Y * 0.7, nextVel.Z)
            bounces = bounces + 1
        end

        table.insert(points, nextPos)
        pos = nextPos
        vel = nextVel

        if vel.Magnitude < 1 then break end
    end

    return points
end

-- Draw trail
local function drawTrail(points)
    TrailFolder:ClearAllChildren()
    for _, point in ipairs(points) do
        local part = Instance.new("Part")
        part.Anchored = true
        part.CanCollide = false
        part.Size = TRAIL_PART_SIZE
        part.Position = point
        part.Color = TRAIL_COLOR
        part.Material = Enum.Material.Neon
        part.Shape = Enum.PartType.Ball
        part.Parent = TrailFolder
    end
end

-- Update trail
RunService.RenderStepped:Connect(function()
    if Ball and Ball.Parent and Ball.Velocity.Magnitude > 1 then
        local points = simulateTrajectoryWithBounces(Ball.Position, Ball.Velocity)
        drawTrail(points)
    end
end)

-- Detect if ball respawns
Workspace.FootballField.ChildAdded:Connect(function(child)
    if child.Name == "SoccerBall" then
        Ball = child
    end
end)
