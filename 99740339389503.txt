local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()

local Window = Rayfield:CreateWindow({
   Name = "",
   Icon = 0, -- Icon in Topbar. Can use Lucide Icons (string) or Roblox Image (number). 0 to use no icon (default).
   LoadingTitle = "",
   LoadingSubtitle = "",
   ShowText = "", -- for mobile users to unhide rayfield, change if you'd like
   Theme = "Default", -- Check https://docs.sirius.menu/rayfield/configuration/themes

   ToggleUIKeybind = "K", Enum.KeyCode)

   DisableRayfieldPrompts = false,
   DisableBuildWarnings = false, has a version mismatch with the interface

   ConfigurationSaving = {
      Enabled = true,
      FolderName = nil, 
      FileName = "Big Hub"
   },

   Discord = {
      Enabled = false, -- Prompt the user to join your Discord server if their executor supports it
      Invite = "noinvitelink", -- The Discord invite code, do not include discord.gg/. E.g. discord.gg/ ABCD would be ABCD
      RememberJoins = true -- Set this to false to make them join the discord every time they load it up
   },

   KeySystem = false, -- Set this to true to use our key system
   KeySettings = {
      Title = "God mode",
      Subtitle = "Key System",
      Note = "No method of obtaining the key is provided", -- Use this to tell the user how to get a key
      FileName = "Key", -- It is recommended to use something unique as other scripts using Rayfield may overwrite your key file
      SaveKey = true, -- The user's key will be saved, but if you change the key, they will be unable to use your script
      GrabKeyFromSite = false, -- If this is true, set Key below to the RAW site you would like Rayfield to get the key from
      Key = {"Hello"} -- List of keys that will be accepted by the system, can be RAW file links (pastebin, github etc) or simple strings ("hello","key22")
   }
})

local Tab = Window:CreateTab("Perks", 4483362458) -- Title, Image

local Section = Tab:CreateSection("Abilities")

Rayfield:Notify({
   Title = "Like the script",
   Content = "Like the script, At some point it will stop working.",
   Duration = 6.5,
   Image = 4483362458,
})

Tab:CreateButton({
	Name = "God mode",
	Callback = function()
		local AbilityEvents = game:GetService("ReplicatedStorage"):WaitForChild("AbilityEvents")
		local remote = AbilityEvents:WaitForChild("LastBreathUsed")

		local duration = 5       
		local interval = 0.3     

		local startTime = tick()

		while tick() - startTime < duration do
			remote:FireServer()
			task.wait(interval)
		end
	end
})


local Button = Tab:CreateButton({
   Name = "Night vision ability",
   Callback = function()
   game:GetService("ReplicatedStorage"):WaitForChild("AbilityEvents"):WaitForChild("NightVisionUsed"):FireServer()
   end,
})

Tab:CreateButton({
	Name = "Ultra Mechanic",
	Callback = function()
		game:GetService("ReplicatedStorage")
			:WaitForChild("AbilityEvents")
			:WaitForChild("UltraMechanicUsed")
			:FireServer()
	end
})

Tab:CreateButton({
	Name = "Safety First",

	end
