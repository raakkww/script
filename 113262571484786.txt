local player = game.Players.LocalPlayer
local AUTO = false
local busy = false


local BLOCK = {"reset","currently","new"}

local function isBlocked(obj)
    local n = obj:GetFullName():lower()
    for _, w in ipairs(BLOCK) do
        if string.find(n, w) then
            return true
        end
    end
    return false
end


local UPGRADES = {}
local BUYABLES = {}

local function collect()
    UPGRADES = {}
    BUYABLES = {}

    local upg = workspace:FindFirstChild("Upgrades")
    local buy = workspace:FindFirstChild("Buyables")

    if upg then
        for _,d in ipairs(upg:GetDescendants()) do
            if d:IsA("ClickDetector") and not isBlocked(d) then
                table.insert(UPGRADES, d)
            end
        end
    end

    if buy then
        for _,d in ipairs(buy:GetDescendants()) do
            if d:IsA("ClickDetector") and not isBlocked(d) then
                table.insert(BUYABLES, d)
            end
        end
    end
end

collect()

-- refresh Upgrade 
task.spawn(function()
    while true do
        task.wait(4)
        collect()
    end
end)

task.spawn(function()
    while true do
        task.wait(0.25)

        if AUTO and not busy then
            busy = true

            -- Nếu reset
            if not player.Character 
            or not player.Character:FindFirstChild("HumanoidRootPart") then

                repeat task.wait()
                until player.Character 
                and player.Character:FindFirstChild("HumanoidRootPart")

                task.wait(0.7) -- delay ổn định
                collect()
            end

            -- Upgrade before
            for i = 1,#UPGRADES do
                if not AUTO then break end
                fireclickdetector(UPGRADES[i])
                task.wait(0.03)
            end

            -- buyable After
            for i = 1,#BUYABLES do
                if not AUTO then break end
                fireclickdetector(BUYABLES[i])
                task.wait(0.03)
            end

            busy = false
        end
    end
end)
