-- [[ HELPER FUNCTIONS ]] --
local function getObject(name, parentName)
    for _, v in pairs(workspace:GetDescendants()) do
        if parentName then
            if v.Name == name and v.Parent and v.Parent.Name == parentName then
                return v
            end
        else
            if v.Name == name then
                return v
            end
        end
    end
    return nil
end

local function bypassTween(targetCFrame)
    if not lp.Character or not lp.Character:FindFirstChild("HumanoidRootPart") then return end
    local hrp = lp.Character.HumanoidRootPart
    local distance = (hrp.Position - targetCFrame.Position).Magnitude
    local speed = 130 
    local info = TweenInfo.new(distance / speed, Enum.EasingStyle.Linear)
    
    local tween = TweenService:Create(hrp, info, {CFrame = targetCFrame})
    tween:Play()
    return tween
end

-- [[ ESP SYSTEM ]] --
local function updateTags(player, hrp, color)
    local tag = hrp:FindFirstChild("SG_Tags")
    if not tag then
        tag = Instance.new("BillboardGui", hrp)
        tag.Name = "SG_Tags"
        tag.Size = UDim2.new(0, 200, 0, 50)
        tag.AlwaysOnTop = true
        tag.StudsOffset = Vector3.new(0, 4, 0)
        local l = Instance.new("TextLabel", tag)
        l.Size = UDim2.new(1, 0, 1, 0)
        l.BackgroundTransparency = 1
        l.TextSize = 14
        l.Font = Enum.Font.SourceSansBold
        l.TextStrokeTransparency = 0
    end
    local label = tag.TextLabel
    label.TextColor3 = color
    local text = ""
    if _G.EspN then text = player.Name end
    if _G.EspD and lp.Character and lp.Character:FindFirstChild("HumanoidRootPart") then
        local d = math.floor((lp.Character.HumanoidRootPart.Position - hrp.Position).Magnitude)
        text = text .. (text ~= "" and " | " or "") .. d .. "m"
    end
    label.Text = text
    label.Visible = (_G.EspN or _G.EspD)
end

task.spawn(function()
    while task.wait(0.5) do
        for _, p in pairs(game.Players:GetPlayers()) do
            if p ~= lp and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
                local hrp = p.Character.HumanoidRootPart
                local isBear = p.Team and (string.find(p.Team.Name, "Bear") or string.find(p.Team.Name, "Urso"))
                
                if _G.EspB and isBear then
                    if not hrp:FindFirstChild("RayBox") then
                        local b = Instance.new("BoxHandleAdornment", hrp)
                        b.Name = "RayBox"; b.Size = Vector3.new(4.5, 6.5, 4.5)
                        b.AlwaysOnTop = true; b.ZIndex = 10; b.Color3 = Color3.new(1,0,0)
                        b.Transparency = 0.6; b.Adornee = hrp
                    end
                    updateTags(p, hrp, Color3.new(1,0,0))
                elseif hrp:FindFirstChild("RayBox") then hrp.RayBox:Destroy() end

                if _G.EspS and not isBear then
                    if not p.Character:FindFirstChild("SHL") then
                        local h = Instance.new("Highlight", p.Character)
                        h.Name = "SHL"; h.FillColor = Color3.fromRGB(0, 160, 255)
                        h.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
                    end
                    updateTags(p, hrp, Color3.fromRGB(0, 160, 255))
                elseif p.Character:FindFirstChild("SHL") then p.Character.SHL:Destroy() end
            end
        end

        for _, o in pairs(workspace:GetDescendants()) do
            if _G.EspAllObj then
                if o.Name == "Wire" and not o:FindFirstChild("ObjHL") then
                    local h = Instance.new("Highlight", o); h.Name = "ObjHL"; h.FillColor = Color3.fromRGB(255, 120, 0); h.DepthMode = "AlwaysOnTop"
                elseif o.Name == "Base" and o.Parent and o.Parent.Name == "CheeseAltar" and not o:FindFirstChild("ObjHL") then
                    local h = Instance.new("Highlight", o); h.Name = "ObjHL"; h.FillColor = Color3.fromRGB(255, 255, 0); h.DepthMode = "AlwaysOnTop"
                end
            elseif not _G.EspAllObj and o:FindFirstChild("ObjHL") then o.ObjHL:Destroy() end

            if _G.EspCheeseItem then
                if (o.Name == "Cheese" or o.Name == "cheese") and (o.Parent.Name == "Items" or o.Parent.Name == "Item") then
                    if not o:FindFirstChild("CheeseHL") then
                        local h = Instance.new("Highlight", o); h.Name = "CheeseHL"; h.FillColor = Color3.fromRGB(0, 0, 0); h.DepthMode = "AlwaysOnTop"
                    end
                end
            elseif not _G.EspCheeseItem and o:FindFirstChild("CheeseHL") then o.CheeseHL:Destroy() end
        end
    end
