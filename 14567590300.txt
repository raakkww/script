-- BACKGROUND LOGIC LOOPS

-- PC Key Controls
UserInputService.InputBegan:Connect(function(input, gp)
    if gp then return end
    if input.KeyCode == Enum.KeyCode.W then pcControls.F = -1
    elseif input.KeyCode == Enum.KeyCode.S then pcControls.B = 1
    elseif input.KeyCode == Enum.KeyCode.A then pcControls.L = -1
    elseif input.KeyCode == Enum.KeyCode.D then pcControls.R = 1
    elseif input.KeyCode == Enum.KeyCode.E then pcControls.U = 1
    elseif input.KeyCode == Enum.KeyCode.Q then pcControls.D = -1 end
end)

UserInputService.InputEnded:Connect(function(input)
    if input.KeyCode == Enum.KeyCode.W then pcControls.F = 0
    elseif input.KeyCode == Enum.KeyCode.S then pcControls.B = 0
    elseif input.KeyCode == Enum.KeyCode.A then pcControls.L = 0
    elseif input.KeyCode == Enum.KeyCode.D then pcControls.R = 0
    elseif input.KeyCode == Enum.KeyCode.E then pcControls.U = 0
    elseif input.KeyCode == Enum.KeyCode.Q then pcControls.D = 0 end
end)

-- Auto Catch Loop (Logic from Source)
task.spawn(function()
    while task.wait(0.05) do
        if _G_DATA.autoCatch then
            local func = ReplicatedStorage:FindFirstChild("Replication") and ReplicatedStorage.Replication:FindFirstChild("Functions") and ReplicatedStorage.Replication.Functions:FindFirstChild("Mom_Catch_Baby_Handler")
            local char = LocalPlayer.Character
            local hrp = char and char:FindFirstChild("HumanoidRootPart")
            if func and char and hrp then
                for _, b in pairs(workspace:GetDescendants()) do
                    if b:IsA("Model") and b:FindFirstChild("Catch_Handler") then
                        local root = b:FindFirstChild("HumanoidRootPart") or b.PrimaryPart
                        if root and (hrp.Position - root.Position).Magnitude <= 12 then 
                            task.spawn(function() pcall(function() func:InvokeServer(b) end) end)
                        end
                    end
                end
            end
        end
    end
end)

-- ESP Loop
task.spawn(function()
    while task.wait(0.5) do
        if _G_DATA.esp then
            for _, plr in pairs(Players:GetPlayers()) do
                if plr ~= LocalPlayer and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
                    pcall(function()
                        local hl = plr.Character:FindFirstChild("KasenHighlight") or Instance.new("Highlight", plr.Character)
                        hl.Name = "KasenHighlight"; hl.FillColor = Color3.fromRGB(120, 0, 255); hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
                        
                        local hrp = plr.Character.HumanoidRootPart
                        if not hrp:FindFirstChild("KasenTag") then
                            local tag = Instance.new("BillboardGui", hrp)
                            tag.Name = "KasenTag"; tag.Size = UDim2.new(0, 200, 0, 50); tag.ExtentsOffset = Vector3.new(0, 3.5, 0); tag.AlwaysOnTop = true
                            local tl = Instance.new("TextLabel", tag)
                            tl.Size = UDim2.new(1, 0, 1, 0); tl.BackgroundTransparency = 1; tl.Text = plr.DisplayName; tl.TextColor3 = Color3.new(1,1,1); tl.Font = Enum.Font.GothamBlack; tl.TextSize = 14
                        end
                    end)
                end
            end
        end
    end
end)

-- Main Render Loop (Movement/Attributes)
RunService.RenderStepped:Connect(function()
    if not LocalPlayer.Character or not LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then return end
    local hrp = LocalPlayer.Character.HumanoidRootPart
    local hum = LocalPlayer.Character.Humanoid
    local char = LocalPlayer.Character

    if _G_DATA.speed then hum.WalkSpeed = _G_DATA.walkSpeed end

    if _G_DATA.noDiveCooldown then
        local stamina = char:FindFirstChild("Dive_Stamina")
        if stamina then stamina.Value = 10 end
        for name, value in pairs(char:GetAttributes()) do
            if string.find(name, "Cooldown") or string.find(name, "Dive") then char:SetAttribute(name, 0) end
        end
    end

    if _G_DATA.flyMobile then
        hum.PlatformStand = true
        local mv = GetMobileMoveVector()
        if mv.Magnitude > 0 then hrp.Velocity = Camera.CFrame:VectorToWorldSpace(mv.Unit) * _G_DATA.flySpeedMobile else hrp.Velocity = Vector3.new(0, 0, 0) end
    elseif _G_DATA.flyPC then
        hum.PlatformStand = true
        local mv = Vector3.new(pcControls.L + pcControls.R, pcControls.U + pcControls.D, pcControls.F + pcControls.B)
        if mv.Magnitude > 0 then hrp.Velocity = Camera.CFrame:VectorToWorldSpace(mv.Unit) * _G_DATA.flySpeedPC else hrp.Velocity = Vector3.new(0, 0, 0) end
    else
        if not _G_DATA.flyMobile and not _G_DATA.flyPC then hum.PlatformStand = false end
    end
end)

-- Collision/Jump
RunService.Stepped:Connect(function()
    if _G_DATA.noclip and LocalPlayer.Character then
        for _, v in pairs(LocalPlayer.Character:GetDescendants()) do if v:IsA("BasePart") then v.CanCollide = false end end
    end
end)

UserInputService.JumpRequest:Connect(function()
    if _G_DATA.infJump and LocalPlayer.Character:FindFirstChild("Humanoid") then
        LocalPlayer.Character.Humanoid:ChangeState(Enum.HumanoidStateType.Jumping)
    end
end)
