local Players = game:GetService("Players")
local UserInputService = game:GetService("UserInputService")
local RunService = game:GetService("RunService")
local TweenService = game:GetService("TweenService")
local LocalPlayer = Players.LocalPlayer

-- === УЛУЧШЕННОЕ ГУИ ===
local ScreenGui = Instance.new("ScreenGui", LocalPlayer:WaitForChild("PlayerGui"))
ScreenGui.Name = "PremiumFarmGui"

local MainFrame = Instance.new("Frame", ScreenGui)
MainFrame.Size = UDim2.new(0, 200, 0, 80)
MainFrame.Position = UDim2.new(0.1, 0, 0.1, 0)
MainFrame.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
MainFrame.BorderSizePixel = 0
MainFrame.Active = true

-- Скругление углов
local UICorner = Instance.new("UICorner", MainFrame)
UICorner.CornerRadius = UDim.new(0, 10)

-- Линия сверху (акцент)
local AccentBar = Instance.new("Frame", MainFrame)
AccentBar.Size = UDim2.new(1, 0, 0, 4)
AccentBar.BackgroundColor3 = Color3.fromRGB(80, 80, 80)
AccentBar.BorderSizePixel = 0
Instance.new("UICorner", AccentBar).CornerRadius = UDim.new(0, 10)

local ToggleBtn = Instance.new("TextButton", MainFrame)
ToggleBtn.Size = UDim2.new(0.9, 0, 0.6, 0)
ToggleBtn.Position = UDim2.new(0.05, 0, 0.25, 0)
ToggleBtn.Text = "PLAYER FARM: OFF"
ToggleBtn.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
ToggleBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
ToggleBtn.Font = Enum.Font.GothamBold
ToggleBtn.TextSize = 14
ToggleBtn.AutoButtonColor = false

local BtnCorner = Instance.new("UICorner", ToggleBtn)
BtnCorner.CornerRadius = UDim.new(0, 8)

-- Тень или обводка кнопки
local UIStroke = Instance.new("UIStroke", ToggleBtn)
UIStroke.Color = Color3.fromRGB(200, 50, 50)
UIStroke.Thickness = 2
UIStroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border

-- Логика перетаскивания (плавная)
local dragStart, startPos, dragging
MainFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true dragStart = input.Position startPos = MainFrame.Position
    end
end)
UserInputService.InputChanged:Connect(function(input)
    if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
        local delta = input.Position - dragStart
        MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)
UserInputService.InputEnded:Connect(function(input) if input.UserInputType == Enum.UserInputType.MouseButton1 then dragging = false end end)

-- === ЛОГИКА ФАРМА ===
local isEnabled = false
local currentTarget = nil
local isLooting = false

ToggleBtn.MouseButton1Click:Connect(function()
    isEnabled = not isEnabled
    
    -- Анимация цвета
    local targetColor = isEnabled and Color3.fromRGB(50, 200, 50) or Color3.fromRGB(200, 50, 50)
    TweenService:Create(UIStroke, TweenInfo.new(0.3), {Color = targetColor}):Play()
    
    ToggleBtn.Text = isEnabled and "PLAYER FARM: ACTIVE" or "PLAYER FARM: OFF"
    if not isEnabled then currentTarget = nil end
end)

-- Поиск цели (раз в 1 сек)
task.spawn(function()
    while true do
        if isEnabled and not currentTarget and not isLooting then
            local closest = nil
            local dist = 500
            if LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
                for _, p in pairs(Players:GetPlayers()) do
                    if p ~= LocalPlayer and p.Character and p.Character:FindFirstChild("HumanoidRootPart") then
                        local hum = p.Character:FindFirstChild("Humanoid")
                        if hum and hum.Health > 0 then
                            local d = (LocalPlayer.Character.HumanoidRootPart.Position - p.Character.HumanoidRootPart.Position).Magnitude
                            if d < dist then dist = d closest = p end
                        end
                    end
                end
            end
            currentTarget = closest
        end
        task.wait(1)
    end
end)

-- Основной цикл (Ниже игрока на 7 единиц + обнуление скорости)
RunService.Heartbeat:Connect(function()
    if isEnabled and currentTarget and not isLooting then
        local char = LocalPlayer.Character
        local root = char and char:FindFirstChild("HumanoidRootPart")
        local tRoot = currentTarget.Character and currentTarget.Character:FindFirstChild("HumanoidRootPart")
        local tHum = currentTarget.Character and currentTarget.Character:FindFirstChild("Humanoid")

        if root and tRoot and tHum then
            if tHum.Health > 0 then
                root.Velocity = Vector3.new(0, 0, 0) 
                root.CFrame = tRoot.CFrame * CFrame.new(0, -7, 0) * CFrame.Angles(math.rad(90), 0, 0)
            else
                isLooting = true
                local dPos = tRoot.CFrame
                currentTarget = nil
                task.spawn(function()
                    for i = 1, 15 do 
                        root.CFrame = dPos
                        if char:FindFirstChild("Humanoid") then char.Humanoid.Jump = true end
                        task.wait(0.5)
                    end
                    isLooting = false
                end)
            end
        else
            currentTarget = nil
        end
    end
end)

-- Удары раз в 40мс
task.spawn(function()
    while true do
        if isEnabled and currentTarget and not isLooting then
            local tool = LocalPlayer.Character:FindFirstChildOfClass("Tool")
            if tool then tool:Activate() end
        end
        task.wait(0.04)
    end
end)
