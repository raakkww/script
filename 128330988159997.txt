local p = game.Players.LocalPlayer

p.DescendantAdded:Connect(function(o)
	if o:GetAttribute("Recoil") then o:SetAttribute("Recoil",0) end
	if o:GetAttribute("Spread") then o:SetAttribute("Spread",0) end
	if o:GetAttribute("FireRate") then o:SetAttribute("FireRate",3000) end
end)

for _,o in pairs(p:GetDescendants()) do
	if o:GetAttribute("Recoil") then o:SetAttribute("Recoil",0) end
	if o:GetAttribute("Spread") then o:SetAttribute("Spread",0) end
	if o:GetAttribute("FireRate") then o:SetAttribute("FireRate",3000) end
end
