local items = workspace:WaitForChild("Items")
local textsize = UDim2.new(0, 200, 0, 50) -- u can change text size

local function CreateEsp(obj)
    if not obj:IsA("Model") or obj:FindFirstChild("ESP_Group") then return end
    
    local group = Instance.new("Folder")
    group.Name = "ESP_Group"
    group.Parent = obj

    local hl = Instance.new("Highlight")
    hl.FillTransparency = 1
    hl.OutlineColor = Color3.fromRGB(255, 255, 255)
    hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
    hl.Adornee = obj
    hl.Parent = group

    -- 횖혶횖째횖쨈횖쩔횖쨍횗혖횗혣 (BillboardGui)
    local bgui = Instance.new("BillboardGui")
    bgui.Name = "NameTag"
    bgui.Adornee = obj:FindFirstChildWhichIsA("BasePart") or obj.PrimaryPart
    bgui.Size = textsize
    bgui.StudsOffset = Vector3.new(0, 2, 0) -- 횖혪횗혢횗혖횖쩐횗혗횖째 횗혗횖쨉횖쨘횗혖횗혗횖째 횖쩍횖째횖쨈 횖쩐횖짹횗혡횖쨉횖쨘횗혗횖쩐횖쩌
    bgui.AlwaysOnTop = true
    bgui.Parent = group

    local tl = Instance.new("TextLabel")
    tl.BackgroundTransparency = 1
    tl.Size = UDim2.new(1, 0, 1, 0)
    tl.Text = obj.Name
    tl.TextColor3 = Color3.new(1, 1, 1)
    tl.TextStrokeTransparency = 0
    tl.TextScaled = true
    tl.Font = Enum.Font.SourceSansBold
    tl.Parent = bgui
end

for _, folder in pairs(items:GetChildren()) do
    for _, v in pairs(folder:GetDescendants()) do
        CreateEsp(v)
    end
    folder.DescendantAdded:Connect(CreateEsp)
end
