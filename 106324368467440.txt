-- Функция для получения размера модели
local function getModelSize(model)
    local parts = model:GetDescendants()
    local min, max = Vector3.new(math.huge, math.huge, math.huge), Vector3.new(-math.huge, -math.huge, -math.huge)
    
    for _, part in pairs(parts) do
        if part:IsA("BasePart") then
            local size = part.Size
            local cframe = part.CFrame
            local corners = {
                cframe * Vector3.new(-size.X/2, -size.Y/2, -size.Z/2),
                cframe * Vector3.new(-size.X/2, -size.Y/2, size.Z/2),
                cframe * Vector3.new(-size.X/2, size.Y/2, -size.Z/2),
                cframe * Vector3.new(-size.X/2, size.Y/2, size.Z/2),
                cframe * Vector3.new(size.X/2, -size.Y/2, -size.Z/2),
                cframe * Vector3.new(size.X/2, -size.Y/2, size.Z/2),
                cframe * Vector3.new(size.X/2, size.Y/2, -size.Z/2),
                cframe * Vector3.new(size.X/2, size.Y/2, size.Z/2)
            }
            for _, corner in pairs(corners) do
                min = Vector3.new(math.min(min.X, corner.X), math.min(min.Y, corner.Y), math.min(min.Z, corner.Z))
                max = Vector3.new(math.max(max.X, corner.X), math.max(max.Y, corner.Y), math.max(max.Z, corner.Z))
            end
        end
    end
    
    if min == Vector3.new(math.huge, math.huge, math.huge) then
        return Vector3.new(2, 2, 2)
    end
    return max - min + Vector3.new(0.2, 0.2, 0.2)
end

-- Функция для получения позиции объекта
local function getObjectPosition(obj)
    if obj:IsA("BasePart") then
        return obj.Position
    elseif obj:IsA("Model") then
        local cframe = obj:GetBoundingBox()
        return cframe.Position
    end
    return Vector3.new(0, 0, 0)
end

-- Функция для создания ESP и надписи
local function createESP(obj)
    if not obj or espElements[obj] then return end

    if not LocalPlayer.Character or not LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
        return
    end
    local playerPos = LocalPlayer.Character.HumanoidRootPart.Position
    local objPos = getObjectPosition(obj)
    local distance = (playerPos - objPos).Magnitude

    if distance > maxRenderDistance then
        return
    end

    local espSize
    if obj:IsA("BasePart") then
        espSize = obj.Size + Vector3.new(0.2, 0.2, 0.2)
    elseif obj:IsA("Model") then
        espSize = getModelSize(obj)
    else
        return
    end

    local isGold = obj.Name == "Gold"
    local isShark = obj.Name == "Shark"

    if (isGold and not showGold) or (isShark and not showShark) then
        return
    end

    local defaultEspColor = isGold and Color3.new(1, 1, 0) or Color3.new(0, 0.5, 1)
    local defaultLabelColor = defaultEspColor
    local defaultLabelText = isGold and "Gold" or "🦈"
    local defaultLabelSize = isGold and UDim2.new(0, 15, 0, 15) or UDim2.new(0, 12.5, 0, 12.5)

    local esp = Instance.new("BoxHandleAdornment")
    esp.Adornee = obj
    esp.AlwaysOnTop = true
    esp.ZIndex = 0
    esp.Size = espSize
    esp.Transparency = 0.5
    esp.Color3 = defaultEspColor
    esp.Parent = CoreGui

    local billboard = Instance.new("BillboardGui")
    billboard.Adornee = obj
    billboard.Size = isGold and UDim2.new(0, 50, 0, 40) or defaultLabelSize
    billboard.StudsOffset = Vector3.new(0, espSize.Y + 1, 0)
    billboard.AlwaysOnTop = true
    billboard.Enabled = true

    local textLabel = Instance.new("TextLabel")
    textLabel.Size = isGold and UDim2.new(1, 0, 0.5, 0) or UDim2.new(1, 0, 1, 0)
    textLabel.Position = isGold and UDim2.new(0, 0, 0, 0) or UDim2.new(0, 0, 0, 0)
    textLabel.BackgroundTransparency = 1
    textLabel.Text = defaultLabelText
    textLabel.TextColor3 = defaultLabelColor
    textLabel.TextScaled = true
    textLabel.Font = Enum.Font.SourceSansBold
    textLabel.Parent = billboard

    local distanceLabel
    if isGold then
        distanceLabel = Instance.new("TextLabel")
        distanceLabel.Size = UDim2.new(1, 0, 0.5, 0)
        distanceLabel.Position = UDim2.new(0, 0, 0.5, 0)
        distanceLabel.BackgroundTransparency = 1
        distanceLabel.Text = math.floor(distance) .. " studs"
        distanceLabel.TextColor3 = defaultLabelColor
        distanceLabel.TextScaled = true
        distanceLabel.Font = Enum.Font.SourceSans
        distanceLabel.Parent = billboard
    end

    billboard.Parent = CoreGui

    espElements[obj] = {
        esp = esp,
        billboard = billboard,
        textLabel = textLabel,
        distanceLabel = distanceLabel,
        defaultSize = espSize,
        defaultLabelSize = defaultLabelSize,
        defaultLabelText = defaultLabelText,
        isGold = isGold,
        defaultEspColor = defaultEspColor,
        defaultLabelColor = defaultLabelColor
    }
end

-- Функция для поиска всех объектов "Gold" и "Shark"
local function findAndApplyESP()
    for _, obj in pairs(workspace:GetDescendants()) do
        if (obj.Name == "Gold" or obj.Name == "Shark") and (obj:IsA("BasePart") or obj:IsA("Model")) then
            createESP(obj)
        end
    end
end

-- Hover effects for buttons
local function addHoverEffect(button)
    local originalColor = button.BackgroundColor3
    button.MouseEnter:Connect(function()
        button.BackgroundColor3 = Color3.fromRGB(
            math.min(originalColor.R * 255 + 20, 255),
            math.min(originalColor.G * 255 + 20, 255),
            math.min(originalColor.B * 255 + 20, 255)
        )
    end)
    button.MouseLeave:Connect(function()
        button.BackgroundColor3 = originalColor
    end)
end

-- Функция телепортации к ближайшему Gold
local function teleportToNearestGold()
    if not LocalPlayer.Character or not LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
        return
    end

    local playerPos = LocalPlayer.Character.HumanoidRootPart.Position
    local nearestGold = nil
    local minDistance = math.huge
    local maxRenderDistance = 650

    for _, obj in pairs(workspace:GetDescendants()) do
        if obj.Name == "Gold" and (obj:IsA("BasePart") or obj:IsA("Model")) then
            local objPos
            if obj:IsA("BasePart") then
                objPos = obj.Position
            elseif obj:IsA("Model") then
                local cframe = obj:GetBoundingBox()
                objPos = cframe.Position
            end
            local distance = (playerPos - objPos).Magnitude
            if distance <= maxRenderDistance and distance < minDistance then
                minDistance = distance
                nearestGold = obj
            end
        end
    end

    if nearestGold then
        local teleportPos
        if nearestGold:IsA("BasePart") then
            teleportPos = nearestGold.Position
        elseif nearestGold:IsA("Model") then
            local cframe = nearestGold:GetBoundingBox()
            teleportPos = cframe.Position
        end
        LocalPlayer.Character.HumanoidRootPart.CFrame = CFrame.new(teleportPos + Vector3.new(0, 5, 0))
    end
end

-- Функция телепортации к BoatOrigin
local function teleportToBoatOrigin()
    local boatOrigin = game.Workspace:FindFirstChild("Boat") and game.Workspace.Boat:FindFirstChild("BoatOrigin")
    if not boatOrigin then
        warn("BoatOrigin не найден в Workspace.Boat!")
        return
    elseif not boatOrigin:IsA("Part") then
        warn("BoatOrigin не является Part! Тип: " .. boatOrigin.ClassName)
        return
    end
    if not LocalPlayer.Character or not LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
        warn("Не удалось телепортировать: персонаж недоступен")
        return
    end
    local position = boatOrigin.Position
    LocalPlayer.Character.HumanoidRootPart.CFrame = CFrame.new(position + Vector3.new(0, 5, 0))
    print("Телепортирован к BoatOrigin (Путь: " .. boatOrigin:GetFullName() .. ")")
end

-- Удаление всех текущих ESP по типу
local function removeESPByType(isGoldType)
    for obj, elements in pairs(espElements) do
        if elements.isGold == isGoldType then
            if elements.esp then pcall(function() elements.esp:Destroy() end) end
            if elements.billboard then pcall(function() elements.billboard:Destroy() end) end
            if elements.distanceLabel then pcall(function() elements.distanceLabel:Destroy() end) end
            if elements.textLabel then pcall(function() elements.textLabel:Destroy() end) end
            espElements[obj] = nil
        end
    end
end

-- Глобальный Heartbeat для обновления всех ESP
RunService.Heartbeat:Connect(function()
    if not LocalPlayer.Character or not LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
        return
    end
    local playerPos = LocalPlayer.Character.HumanoidRootPart.Position
    
    for obj, elements in pairs(espElements) do
        if not obj or not obj:IsDescendantOf(workspace) then
            if elements.esp then pcall(function() elements.esp:Destroy() end) end
            if elements.billboard then pcall(function() elements.billboard:Destroy() end) end
            if elements.distanceLabel then pcall(function() elements.distanceLabel:Destroy() end) end
            if elements.textLabel then pcall(function() elements.textLabel:Destroy() end) end
            espElements[obj] = nil
            continue
        end
        
        local objPos = getObjectPosition(obj)
        local distance = (playerPos - objPos).Magnitude
        
        if elements.isGold and not showGold then
            if elements.esp then pcall(function() elements.esp:Destroy() end) end
            if elements.billboard then pcall(function() elements.billboard:Destroy() end) end
            if elements.distanceLabel then pcall(function() elements.distanceLabel:Destroy() end) end
            if elements.textLabel then pcall(function() elements.textLabel:Destroy() end) end
            espElements[obj] = nil
            continue
        elseif not elements.isGold and not showShark then
            if elements.esp then pcall(function() elements.esp:Destroy() end) end
            if elements.billboard then pcall(function() elements.billboard:Destroy() end) end
            if elements.distanceLabel then pcall(function() elements.distanceLabel:Destroy() end) end
            if elements.textLabel then pcall(function() elements.textLabel:Destroy() end) end
            espElements[obj] = nil
            continue
        end
        
        if distance > maxRenderDistance then
            elements.esp.Transparency = 1
            elements.billboard.Enabled = false
            continue
        else
            elements.esp.Transparency = 0.5
            elements.billboard.Enabled = true
        end
        
        if elements.isGold then
            if elements.distanceLabel then
                elements.distanceLabel.Text = math.floor(distance) .. " studs"
                if distance <= 150 then
                    elements.esp.Color3 = Color3.new(1, 0, 0)
                    elements.esp.Size = elements.defaultSize * 1.1
                    elements.distanceLabel.TextColor3 = Color3.new(1, 0, 0)
                    if elements.textLabel then elements.textLabel.TextColor3 = Color3.new(1, 0, 0) end
                else
                    elements.esp.Color3 = elements.defaultEspColor
                    elements.esp.Size = elements.defaultSize
                    elements.distanceLabel.TextColor3 = elements.defaultLabelColor
                    if elements.textLabel then elements.textLabel.TextColor3 = elements.defaultLabelColor end
                end
            end
        else
            if distance <= 100 then
                if elements.textLabel then
                    elements.textLabel.Text = "♦️"
                    elements.textLabel.TextColor3 = Color3.new(1, 0, 0)
                end
                elements.billboard.Size = UDim2.new(0, 41.67, 0, 41.67)
                elements.esp.Size = elements.defaultSize * 1.1
            else
                if elements.textLabel then
                    elements.textLabel.Text = elements.defaultLabelText
                    elements.textLabel.TextColor3 = elements.defaultLabelColor
                end
                elements.billboard.Size = elements.defaultLabelSize
                elements.esp.Size = elements.defaultSize
            end
            elements.esp.Color3 = elements.defaultEspColor
        end
    end
end)
