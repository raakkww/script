local function randWait(min, max)
    task.wait(min + (math.random() * (max - min)))
end

local function isHighlighted(char)
    if not char then return false end
    if char:FindFirstChildOfClass("Highlight") then return true end
    if char:FindFirstChildOfClass("SelectionBox") then return true end
    for _, v in ipairs(char:GetChildren()) do
        if v:IsA("Folder") and v:FindFirstChildWhichIsA("Highlight") then return true end
    end
    return false
end

local function getTagged()
    for _, p in ipairs(plrs:GetPlayers()) do
        if p ~= lp and p.Character and isHighlighted(p.Character) then
            return p
        end
    end
    return nil
end

local function getTaggedRoot()
    local p = getTagged()
    return p and p.Character and p.Character:FindFirstChild("HumanoidRootPart")
end

local function safeTeleport(target)
    local root = lp.Character and lp.Character:FindFirstChild("HumanoidRootPart")
    if not root then return end
    local dist = (root.Position - target.Position).Magnitude
    local time = math.clamp(dist / 500, 0.05, 0.3)
    local tween = ts:Create(root, TweenInfo.new(time, Enum.EasingStyle.Linear), {CFrame = target})
    tween:Play()
end

local function tpBackLater(time)
    local char = lp.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    local target = getTaggedRoot()
    if root and target then
        local old = root.CFrame
        safeTeleport(target.CFrame * CFrame.new(0,3,0))
        task.wait(time or 1)
        if root and root.Parent then
            safeTeleport(old)
        end
    end
end

local function randomTP()
    local root = lp.Character and lp.Character:FindFirstChild("HumanoidRootPart")
    if root then
        local pos = root.Position + Vector3.new(math.random(-50,50), 3, math.random(-50,50))
        safeTeleport(CFrame.new(pos))
    end
end

local function makeChair(pos)
    if chair and chair.Parent then chair:Destroy() end
    chair = Instance.new("Part")
    chair.Name = "chair"
    chair.Anchored = true
    chair.CanCollide = false
    chair.Transparency = 1
    chair.Size = Vector3.new(2,1,2)
    chair.Position = pos
    chair.Parent = workspace
    task.delay(2, function()
        if chair and chair.Parent then chair:Destroy() end
    end)
    return chair
end

local function spinToggle(state)
    if spinLoop then spinLoop:Disconnect() spinLoop = nil end
    if state then
        spinLoop = rs.RenderStepped:Connect(function()
            local root = lp.Character and lp.Character:FindFirstChild("HumanoidRootPart")
            if root then
                root.CFrame = root.CFrame * CFrame.Angles(0, math.rad(cfg.SpinSpd), 0)
            end
        end)
    end
end

local function velocityFling(plr)
    local char = lp.Character
    local hum = char and char:FindFirstChildOfClass("Humanoid")
    local root = hum and hum.RootPart
    local tChar = plr and plr.Character
    local tRoot = tChar and tChar:FindFirstChild("HumanoidRootPart")
    
    if not (char and hum and root and tChar and tRoot) then return end
    
    local oldPos = root.CFrame
    local oldHeight = workspace.FallenPartsDestroyHeight
    workspace.FallenPartsDestroyHeight = -5000
    
    local bv = Instance.new("BodyVelocity")
    bv.Parent = root
    bv.Velocity = Vector3.new(0,0,0)
    bv.MaxForce = Vector3.new(5000,5000,5000)
    
    local bg = Instance.new("BodyGyro")
    bg.Parent = root
    bg.MaxTorque = Vector3.new(5000,5000,5000)
    bg.P = 1000
    bg.CFrame = tRoot.CFrame * CFrame.Angles(math.rad(90),0,0)
    
    hum:SetStateEnabled(Enum.HumanoidStateType.Seated, false)
    cam.CameraSubject = tRoot
    
    for i = 1, 40 do
        if root and root.Parent then
            local dir = (tRoot.Position - root.Position).Unit
            root.Velocity = dir * 150 + Vector3.new(0,50,0)
            root.RotVelocity = Vector3.new(5000,5000,5000)
            randWait(0.01, 0.03)
        end
    end
    
    bv:Destroy()
    bg:Destroy()
    hum:SetStateEnabled(Enum.HumanoidStateType.Seated, true)
    cam.CameraSubject = hum
    safeTeleport(oldPos)
    workspace.FallenPartsDestroyHeight = oldHeight
end

    Title = "Auto Play (BETA)",
    Desc = "chase tagged, run if you're it",
    Value = cfg.AutoPlay,
    Callback = function(v)
        cfg.AutoPlay = v
        if v then
            if conn.AutoPlay then conn.AutoPlay:Disconnect() end
            conn.AutoPlay = task.spawn(function()
                while cfg.AutoPlay do
                    if not isPathing then
                        isPathing = true
                        local char = lp.Character
                        local root = char and char:FindFirstChild("HumanoidRootPart")
                        local hum = char and char:FindFirstChildOfClass("Humanoid")
                        if root and hum then
                            if isHighlighted(char) then
                                local away = root.Position * -1
                                hum:MoveTo(Vector3.new(away.X, root.Position.Y, away.Z))
                            else
                                local target = getTaggedRoot()
                                if target then
                                    local success, err = pcall(function()
                                        local path = ps:CreatePath({AgentCanJump = true})
                                        path:ComputeAsync(root.Position, target.Position)
                                        if path.Status == Enum.PathStatus.Success then
                                            local wp = path:GetWaypoints()
                                            for i = 1, math.min(3, #wp) do
                                                if wp[i].Action == Enum.PathWaypointAction.Jump then hum.Jump = true end
                                                hum:MoveTo(wp[i].Position)
                                                hum.MoveToFinished:Wait(0.5)
                                            end
                                        else
                                            hum:MoveTo(target.Position)
                                        end
                                    end)
                                end
                            end
                        end
                        isPathing = false
                    end
                    randWait(0.4, 0.6)
                end
            end)
        elseif conn.AutoPlay then
            task.cancel(conn.AutoPlay)
            conn.AutoPlay = nil
        end
    end
})

local function onChar(char)
    task.wait(0.5)
    local hum = char:FindFirstChildOfClass("Humanoid")
    if hum then
        hum.WalkSpeed = cfg.Speed
        hum.JumpPower = cfg.Jump
    end
    if cfg.Invis then
        for _, p in ipairs(char:GetDescendants()) do
            if p:IsA("BasePart") and p.Name ~= "HumanoidRootPart" then
                p.LocalTransparencyModifier = 0.8
            end
        end
    end
end
