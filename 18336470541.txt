-- =========================
-- HITBOX EXPANDER (FIXED)
-- =========================

local Hitbox = {
	Enabled = false,
	Size = 6,
	Part = "HumanoidRootPart",
	Cache = {}
}

local function applyHitbox(char)
	if not char or char == LocalPlayer.Character then return end

	local hum = char:FindFirstChildOfClass("Humanoid")
	local part = char:FindFirstChild(Hitbox.Part)
	if not hum or hum.Health <= 0 or not part then return end

	if not Hitbox.Cache[part] then
		Hitbox.Cache[part] = {
			Size = part.Size,
			CanCollide = part.CanCollide,
		}
	end

	part.Size = Vector3.new(Hitbox.Size, Hitbox.Size, Hitbox.Size)
	part.Transparency = 1
	part.CanCollide = false

	local box = part:FindFirstChild("HitboxOutline") or Instance.new("SelectionBox", part)
	box.Name = "HitboxOutline"
	box.Adornee = part
	box.LineThickness = 0.05
	box.Color3 = Color3.fromRGB(255,255,255)
	box.SurfaceTransparency = 0.6
end

local function restoreHitbox(char)
	local part = char and char:FindFirstChild(Hitbox.Part)
	if not part then return end

	local c = Hitbox.Cache[part]
	if not c then return end

	part.Size = c.Size
	part.CanCollide = c.CanCollide
	part.Transparency = 0

	if part:FindFirstChild("HitboxOutline") then
		part.HitboxOutline:Destroy()
	end

	Hitbox.Cache[part] = nil
end

local function updateHitboxes()
	for _, p in ipairs(Players:GetPlayers()) do
		if p ~= LocalPlayer and p.Character then
			if Hitbox.Enabled then
				applyHitbox(p.Character)
			else
				restoreHitbox(p.Character)
			end
		end
	end
end

local function onCharacterAdded(char)
	char:WaitForChild("HumanoidRootPart", 5)
	if Hitbox.Enabled then
		task.wait(0.3)
		applyHitbox(char)
	end
end

for _, p in ipairs(Players:GetPlayers()) do
	if p.Character then
		task.spawn(onCharacterAdded, p.Character)
	end
	p.CharacterAdded:Connect(onCharacterAdded)
end

Players.PlayerAdded:Connect(function(p)
	p.CharacterAdded:Connect(onCharacterAdded)
end)

RunService.PreSimulation:Connect(function()
	local char = LocalPlayer.Character
	if not char then return end
	local hum = char:FindFirstChildOfClass("Humanoid")
	local hrp = char:FindFirstChild("HumanoidRootPart")
	if not hum or not hrp then return end

	if State.Speed and hum.MoveDirection.Magnitude > 0 and not State.Freeze then
		hrp.CFrame += hum.MoveDirection * State.SpeedValue
	end

	if State.Noclip then
		for _, v in ipairs(char:GetDescendants()) do
			if v:IsA("BasePart") then
				v.CanCollide = false
			end
		end
	end
end)

-- =========================
-- ESP (UNCHANGED)
-- =========================

local SAFE_WEAPONS = {
	"usp","glock","p2000","p250","five","tec","cz","deagle",
	"revolver","dual","knife","dagger","pistol"
}

local function clearESP()
	for _, p in ipairs(Players:GetPlayers()) do
		if p.Character then
			if p.Character:FindFirstChild("WeaponHighlight") then p.Character.WeaponHighlight:Destroy() end
			if p.Character:FindFirstChild("WeaponInfo") then p.Character.WeaponInfo:Destroy() end
		end
	end
end

RunService.Heartbeat:Connect(function()
	if not State.ESP then return end

	for _, p in ipairs(Players:GetPlayers()) do
		if p ~= LocalPlayer and p.Character then
			local char = p.Character
			local hum = char:FindFirstChildOfClass("Humanoid")
			local hrp = char:FindFirstChild("HumanoidRootPart")
			if not hum or not hrp then continue end

			local tool = char:FindFirstChildOfClass("Tool")
			local weaponName = tool and tool.Name or "SAFE"
			local isGun = false

			if tool then
				isGun = true
				for _, s in ipairs(SAFE_WEAPONS) do
					if weaponName:lower():find(s) then
						isGun = false
						break
					end
				end
			end

			local hl = char:FindFirstChild("WeaponHighlight") or Instance.new("Highlight", char)
			hl.Name = "WeaponHighlight"
			hl.FillTransparency = 1
			hl.OutlineColor = isGun and Color3.fromRGB(255,0,0) or Color3.fromRGB(0,255,0)
			hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop

			local gui = char:FindFirstChild("WeaponInfo") or Instance.new("BillboardGui", char)
			gui.Name = "WeaponInfo"
			gui.Size = UDim2.new(0,200,0,70)
			gui.StudsOffset = Vector3.new(0,4,0)
			gui.AlwaysOnTop = true

			local label = gui:FindFirstChild("Text") or Instance.new("TextLabel", gui)
			label.Name = "Text"
			label.Size = UDim2.new(1,0,1,0)
			label.BackgroundTransparency = 1
			label.Font = Enum.Font.GothamBold
			label.TextSize = 13
			label.RichText = true
			label.TextColor3 = Color3.new(1,1,1)

			local dist = math.floor((LocalPlayer.Character.HumanoidRootPart.Position - hrp.Position).Magnitude)

			label.Text = string.format(
				'<font color="rgb(255,255,255)">%s</font>\n<font color="%s">HP: %d</font>\n<font color="rgb(200,200,200)">%s</font>\n<font color="rgb(170,85,255)">[%dm]</font>',
				p.Name,
				hum.Health < 30 and "rgb(255,0,0)" or "rgb(0,255,0)",
				math.floor(hum.Health),
				weaponName,
				dist
			)
		end
	end
end)

local savedCF, platform
local function toggleAutoWin(on)
	local hrp = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
	if not hrp then return end

	if on then
		savedCF = hrp.CFrame
		hrp.CFrame += Vector3.new(0,-50,0)
		platform = Instance.new("Part", workspace)
		platform.Size = Vector3.new(20,1,20)
		platform.Anchored = true
		platform.Transparency = 0.5
		platform.CFrame = hrp.CFrame + Vector3.new(0,-3.5,0)
	else
		if platform then platform:Destroy() platform=nil end
		if savedCF then hrp.CFrame = savedCF end
	end
end

	State.Freeze = v
	if LocalPlayer.Character then
		LocalPlayer.Character.HumanoidRootPart.Anchored = v
	end
end)

Toggles.ESP:OnChanged(function(v) State.ESP = v if not v then clearESP() end end)
Toggles.Fullbright:OnChanged(function(v) Lighting.Brightness = v and 2 or 1 end)

Toggles.AutoWin:OnChanged(function(v) toggleAutoWin(v) end)

Toggles.HitboxExpander:OnChanged(function(v)
	Hitbox.Enabled = v
	updateHitboxes()
end)

Options.HitboxSize:OnChanged(function()
	Hitbox.Size = Options.HitboxSize.Value
	if Hitbox.Enabled then updateHitboxes() end
end)
