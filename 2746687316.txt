function updateESP()
    for _, v in pairs(_Workspace:GetDescendants()) do
        if v.Name == 'ESP' then v:Destroy() end
    end
    if getgenv().ESPEnabled then
        local models = _Workspace:FindFirstChild("Playermodels")
        if models then
            for _, v in pairs(models:GetChildren()) do
                createESP(v)
            end
        end
    end
end
