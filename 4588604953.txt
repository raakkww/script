local Players = game:GetService("Players")
local UIS = game:GetService("UserInputService")
local RunService = game:GetService("RunService")
local Lighting = game:GetService("Lighting")
local GuiService = game:GetService("GuiService")
local LocalPlayer = Players.LocalPlayer
local PlayerGui = LocalPlayer:WaitForChild("PlayerGui")
local Mouse = LocalPlayer:GetMouse()
local Camera = workspace.CurrentCamera

local CORRECT_KEY = "KUTAK-2026"
local DISCORD_LINK = "https://discord.gg/6R5V8QKYRG"

if PlayerGui:FindFirstChild("KutakWare_Final") then PlayerGui:FindFirstChild("KutakWare_Final"):Destroy() end

local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Name = "KutakWare_Final"; ScreenGui.ResetOnSpawn = false; ScreenGui.Parent = PlayerGui

-- Login Window
local KeyFrame = Instance.new("Frame", ScreenGui)
KeyFrame.Size = UDim2.new(0, 350, 0, 220); KeyFrame.Position = UDim2.new(0.5, -175, 0.4, 0)
KeyFrame.BackgroundColor3 = Color3.fromRGB(20, 20, 20); Instance.new("UICorner", KeyFrame)

local KeyTitle = Instance.new("TextLabel", KeyFrame)
KeyTitle.Size = UDim2.new(1, 0, 0, 50); KeyTitle.Text = "🔑 KUTAK WARE | LOGIN"; KeyTitle.TextColor3 = Color3.new(1,1,1); KeyTitle.BackgroundTransparency = 1; KeyTitle.Font = Enum.Font.RobotoMono; KeyTitle.TextSize = 18

local KeyInput = Instance.new("TextBox", KeyFrame)
KeyInput.Size = UDim2.new(0.8, 0, 0, 45); KeyInput.Position = UDim2.new(0.1, 0, 0.35, 0); KeyInput.PlaceholderText = "Enter Key..."; KeyInput.Text = ""; KeyInput.TextSize = 18; KeyInput.Font = Enum.Font.RobotoMono
KeyInput.BackgroundColor3 = Color3.fromRGB(35, 35, 35); KeyInput.TextColor3 = Color3.new(1,1,1); Instance.new("UICorner", KeyInput)

local CheckBtn = Instance.new("TextButton", KeyFrame)
CheckBtn.Size = UDim2.new(0.38, 0, 0, 45); CheckBtn.Position = UDim2.new(0.1, 0, 0.65, 0); CheckBtn.Text = "Login"; CheckBtn.BackgroundColor3 = Color3.fromRGB(0, 120, 0); CheckBtn.TextColor3 = Color3.new(1,1,1); CheckBtn.TextSize = 18; CheckBtn.Font = Enum.Font.RobotoMono; Instance.new("UICorner", CheckBtn)

local GetBtn = Instance.new("TextButton", KeyFrame)
GetBtn.Size = UDim2.new(0.38, 0, 0, 45); GetBtn.Position = UDim2.new(0.52, 0, 0.65, 0); GetBtn.Text = "Discord"; GetBtn.BackgroundColor3 = Color3.fromRGB(88, 101, 242); GetBtn.TextColor3 = Color3.new(1,1,1); GetBtn.TextSize = 18; GetBtn.Font = Enum.Font.RobotoMono; Instance.new("UICorner", GetBtn)

local function StartScript()
    KeyFrame:Destroy()
    
    local Config = {
        ESP = false, Aimlock = false, FullBright = false, AutoLockpick = false, WallCheck = true,
        AimPart = "Head", Smoothness = 0.15, AimFOV = 150, CameraFOV = 70
    }
    
    local OrigLight = {B = Lighting.Brightness, C = Lighting.ClockTime, A = Lighting.Ambient, S = Lighting.GlobalShadows}
    
    local MainFrame = Instance.new("Frame", ScreenGui)
    MainFrame.Size = UDim2.new(0, 500, 0, 450); MainFrame.Position = UDim2.new(0.5, -250, 0.2, 0)
    MainFrame.BackgroundColor3 = Color3.fromRGB(15, 15, 15); Instance.new("UICorner", MainFrame)

    local Header = Instance.new("Frame", MainFrame)
    Header.Size = UDim2.new(1, 0, 0, 60); Header.BackgroundColor3 = Color3.fromRGB(25, 25, 25); Instance.new("UICorner", Header)
    local Title = Instance.new("TextLabel", Header)
    Title.Size = UDim2.new(1, 0, 1, 0); Title.BackgroundTransparency = 1; Title.Text = "🛡️ KUTAK WARE | ACTIVE"; Title.TextColor3 = Color3.new(1,1,1); Title.Font = Enum.Font.RobotoMono; Title.TextSize = 18

    local LeftSide = Instance.new("Frame", MainFrame); LeftSide.Size = UDim2.new(0.5, -15, 1, -80); LeftSide.Position = UDim2.new(0, 10, 0, 70); LeftSide.BackgroundTransparency = 1
    local RightSide = Instance.new("Frame", MainFrame); RightSide.Size = UDim2.new(0.5, -15, 1, -80); RightSide.Position = UDim2.new(0.5, 5, 0, 70); RightSide.BackgroundTransparency = 1

    local function updateBtnVisual(btn, state)
        btn.BackgroundColor3 = state and Color3.fromRGB(40, 120, 40) or Color3.fromRGB(30, 30, 30)
    end

    local function createBtn(text, parent, order, cb, defaultState)
        local btn = Instance.new("TextButton", parent); btn.Size = UDim2.new(1, 0, 0, 40); btn.Position = UDim2.new(0, 0, 0, (order-1)*46); btn.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
        btn.Text = text; btn.TextColor3 = Color3.new(1,1,1); btn.Font = Enum.Font.RobotoMono; btn.TextSize = 14; Instance.new("UICorner", btn)
        if defaultState ~= nil then updateBtnVisual(btn, defaultState) end
        btn.MouseButton1Click:Connect(function() cb(btn) end) return btn
    end

    createBtn("🧊 2D ESP: [OFF]", LeftSide, 1, function(b) 
        Config.ESP = not Config.ESP; b.Text = "🧊 2D ESP: [" .. (Config.ESP and "ON" or "OFF") .. "]"; updateBtnVisual(b, Config.ESP) 
    end, false)
    
    createBtn("🎯 AIMLOCK: [OFF]", LeftSide, 2, function(b) 
        Config.Aimlock = not Config.Aimlock; b.Text = "🎯 AIMLOCK: [" .. (Config.Aimlock and "ON" or "OFF") .. "]"; updateBtnVisual(b, Config.Aimlock) 
    end, false)

    createBtn("🧱 WALL CHECK: [ON]", LeftSide, 3, function(b) 
        Config.WallCheck = not Config.WallCheck; b.Text = "🧱 WALL CHECK: [" .. (Config.WallCheck and "ON" or "OFF") .. "]"; updateBtnVisual(b, Config.WallCheck) 
    end, true)

    createBtn("☀️ BRIGHT: [OFF]", LeftSide, 4, function(b) 
        Config.FullBright = not Config.FullBright; b.Text = "☀️ BRIGHT: [" .. (Config.FullBright and "ON" or "OFF") .. "]"; updateBtnVisual(b, Config.FullBright)
        if not Config.FullBright then Lighting.Brightness = OrigLight.B; Lighting.ClockTime = OrigLight.C; Lighting.Ambient = OrigLight.A; Lighting.GlobalShadows = OrigLight.S end
    end, false)

    createBtn("🎯 TARGET: HEAD", LeftSide, 5, function(b) 
        Config.AimPart = (Config.AimPart == "Head") and "HumanoidRootPart" or "Head"; b.Text = "🎯 TARGET: " .. (Config.AimPart == "Head" and "HEAD" or "TORSO") 
    end)

    local function createSlider(name, parent, order)
        local f = Instance.new("Frame", parent); f.Size = UDim2.new(1, 0, 0, 70); f.Position = UDim2.new(0, 0, 0, (order-1)*78); f.BackgroundColor3 = Color3.fromRGB(25, 25, 25); Instance.new("UICorner", f)
        local l = Instance.new("TextLabel", f); l.Size = UDim2.new(1, 0, 0, 35); l.BackgroundTransparency = 1; l.Text = name; l.TextColor3 = Color3.new(1,1,1); l.Font = Enum.Font.RobotoMono; l.TextSize = 13
        local b = Instance.new("Frame", f); b.Size = UDim2.new(0.85, 0, 0, 5); b.Position = UDim2.new(0.075, 0, 0.75, 0); b.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
        local d = Instance.new("Frame", b); d.Size = UDim2.new(0, 14, 0, 14); d.Position = UDim2.new(0.3, -7, 0.5, -7); d.BackgroundColor3 = Color3.new(1,1,1); Instance.new("UICorner", d).CornerRadius = UDim.new(1,0)
        local s = false
        f.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then s = true end end)
        UIS.InputEnded:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then s = false end end)
        return {bar = b, dot = d, label = l, getSliding = function() return s end}
    end

    local aimFovS = createSlider("⚙️ AIM FOV", RightSide, 1)
    local smoothS = createSlider("⚙️ SMOOTHNESS", RightSide, 2)
    local camFovS = createSlider("⚙️ CAM FOV", RightSide, 3)

    -- ESP Logic (Bottom-up Health bar)
    local ESP_Objects = {}
    local function CreateDrawingESP(player)
        if player == LocalPlayer then return end
        local Box = Drawing.new("Square"); Box.Thickness = 1; Box.Color = Color3.new(1, 1, 1); Box.Visible = false
        local HealthBar = Drawing.new("Square"); HealthBar.Thickness = 1; HealthBar.Filled = true; HealthBar.Visible = false
        local Name = Drawing.new("Text"); Name.Size = 14; Name.Center = true; Name.Outline = true; Name.Color = Color3.new(1,1,1); Name.Visible = false
        ESP_Objects[player] = {Box = Box, HealthBar = HealthBar, Name = Name}
        
        RunService.RenderStepped:Connect(function()
            local char = player.Character; local hum = char and char:FindFirstChildOfClass("Humanoid"); local hrp = char and char:FindFirstChild("HumanoidRootPart")
            if Config.ESP and char and hum and hrp and hum.Health > 0 then
                local pos, onS = Camera:WorldToViewportPoint(hrp.Position)
                if onS then
                    local sizeY = (Camera:WorldToViewportPoint(hrp.Position + Vector3.new(0, 3, 0)).Y - Camera:WorldToViewportPoint(hrp.Position - Vector3.new(0, 3, 0)).Y)
                    local sizeX = sizeY / 2; local boxX, boxY = pos.X - (sizeX/2), pos.Y - (sizeY/2)
                    Box.Position = Vector2.new(boxX, boxY); Box.Size = Vector2.new(sizeX, sizeY); Box.Visible = true
                    
                    local hpPct = math.clamp(hum.Health / hum.MaxHealth, 0, 1)
                    HealthBar.Size = Vector2.new(3, sizeY * hpPct)
                    HealthBar.Position = Vector2.new(boxX - 6, (boxY + sizeY) - (sizeY * hpPct))
                    HealthBar.Color = Color3.fromHSV(hpPct * 0.35, 1, 1); HealthBar.Visible = true
                    
                    Name.Position = Vector2.new(pos.X, boxY - 18); Name.Text = player.Name; Name.Visible = true; return
                end
            end
            Box.Visible = false; HealthBar.Visible = false; Name.Visible = false
        end)
    end
    for _, p in pairs(Players:GetPlayers()) do CreateDrawingESP(p) end; Players.PlayerAdded:Connect(CreateDrawingESP)

    -- Aimlock & Wall Check Core
    local FOVCircle = Drawing.new("Circle"); FOVCircle.Thickness = 1; FOVCircle.Color = Color3.new(1,1,1); FOVCircle.Visible = false
    
    local function IsVisible(targetPart)
        if not Config.WallCheck then return true end
        local result = workspace:Raycast(Camera.CFrame.Position, targetPart.Position - Camera.CFrame.Position, RaycastParams.new())
        local ignoreList = {LocalPlayer.Character, targetPart.Parent}
        local params = RaycastParams.new(); params.FilterType = Enum.RaycastFilterType.Exclude; params.FilterDescendantsInstances = ignoreList
        local ray = workspace:Raycast(Camera.CFrame.Position, targetPart.Position - Camera.CFrame.Position, params)
        return ray == nil
    end

    RunService.RenderStepped:Connect(function()
        if Config.FullBright then Lighting.Brightness = 2; Lighting.ClockTime = 14; Lighting.Ambient = Color3.new(1,1,1); Lighting.GlobalShadows = false end
        
        if aimFovS.getSliding() then
            local x = math.clamp((Mouse.X - aimFovS.bar.AbsolutePosition.X) / aimFovS.bar.AbsoluteSize.X, 0, 1)
            aimFovS.dot.Position = UDim2.new(x, -7, 0.5, -7); Config.AimFOV = math.floor(x * 500) + 50; aimFovS.label.Text = "⚙️ AIM FOV: " .. Config.AimFOV
        elseif smoothS.getSliding() then
            local x = math.clamp((Mouse.X - smoothS.bar.AbsolutePosition.X) / smoothS.bar.AbsoluteSize.X, 0, 1)
            smoothS.dot.Position = UDim2.new(x, -7, 0.5, -7); Config.Smoothness = (x * 0.95) + 0.05; smoothS.label.Text = "⚙️ SMOOTHNESS: " .. string.format("%.2f", Config.Smoothness)
        elseif camFovS.getSliding() then
            local x = math.clamp((Mouse.X - camFovS.bar.AbsolutePosition.X) / camFovS.bar.AbsoluteSize.X, 0, 1)
            camFovS.dot.Position = UDim2.new(x, -7, 0.5, -7); Config.CameraFOV = math.floor(x * 50) + 70; camFovS.label.Text = "⚙️ CAM FOV: " .. Config.CameraFOV; Camera.FieldOfView = Config.CameraFOV
        end

        FOVCircle.Radius = Config.AimFOV; FOVCircle.Position = Vector2.new(Mouse.X, Mouse.Y + GuiService:GetGuiInset().Y)
        FOVCircle.Visible = Config.Aimlock 

        if Config.Aimlock and UIS:IsMouseButtonPressed(Enum.UserInputType.MouseButton2) then
            local target, dist = nil, Config.AimFOV
            for _, v in pairs(Players:GetPlayers()) do
                if v ~= LocalPlayer and v.Character and v.Character:FindFirstChild(Config.AimPart) then
                    local hum = v.Character:FindFirstChildOfClass("Humanoid")
                    if hum and hum.Health > 0 then
                        local part = v.Character[Config.AimPart]
                        local pos, onS = Camera:WorldToViewportPoint(part.Position)
                        if onS and IsVisible(part) then
                            local mDist = (Vector2.new(pos.X, pos.Y) - Vector2.new(Mouse.X, Mouse.Y)).Magnitude
                            if mDist < dist then dist = mDist; target = part end
                        end
                    end
                end
            end
            if target then Camera.CFrame = Camera.CFrame:Lerp(CFrame.new(Camera.CFrame.Position, target.Position), Config.Smoothness) end
        end
    end)

    local dragT, dragS, startP
    Header.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then dragT = true; dragS = i.Position; startP = MainFrame.Position end end)
    UIS.InputChanged:Connect(function(i) if dragT and i.UserInputType == Enum.UserInputType.MouseMovement then local delta = i.Position - dragS; MainFrame.Position = UDim2.new(startP.X.Scale, startP.X.Offset + delta.X, startP.Y.Scale, startP.Y.Offset + delta.Y) end end)
    UIS.InputEnded:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then dragT = false end end)
    UIS.InputBegan:Connect(function(i, g) if not g and i.KeyCode == Enum.KeyCode.K then MainFrame.Visible = not MainFrame.Visible end end)
end

CheckBtn.MouseButton1Click:Connect(function()
    if KeyInput.Text == CORRECT_KEY then StartScript() else CheckBtn.Text = "Invalid Key"; task.wait(1); CheckBtn.Text = "Login" end
end)
GetBtn.MouseButton1Click:Connect(function() if setclipboard then setclipboard(DISCORD_LINK) end; GetBtn.Text = "Copied"; task.wait(1); GetBtn.Text = "Discord" end)
