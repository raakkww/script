-- Helper functions
local function colorMatches(colorSeq, targetColor)
	local keypoints = colorSeq.Keypoints
	if #keypoints > 0 then
		local c = keypoints[1].Value
		return math.abs(c.R - targetColor.R) < 0.02
		and math.abs(c.G - targetColor.G) < 0.02
		and math.abs(c.B - targetColor.B) < 0.02
	end
	return false
end

local function clearAllOres()
	-- Stop live updates
	if liveUpdateConnection then
		liveUpdateConnection:Disconnect()
		liveUpdateConnection = nil
	end
	
	-- Clear all tracked ores
	for _, oreData in ipairs(trackedOres) do
		if oreData.Highlight and oreData.Highlight.Parent then
			oreData.Highlight:Destroy()
		end
		if oreData.Billboard and oreData.Billboard.Parent then
			oreData.Billboard:Destroy()
		end
	end
	table.clear(trackedOres)
	
	-- Clear radar dots
	for _, dot in ipairs(radarDots) do
		if dot and dot.Parent then
			dot:Destroy()
		end
	end
	table.clear(radarDots)
	
	-- Clear arrow
	if arrow3D then
		arrow3D:Destroy()
		arrow3D = nil
	end
end

-- MAIN SCAN FUNCTION
local function scanOres()
	if not isScanning then return end
	
	-- Clear previous
	clearAllOres()
	
	local character = player.Character
	if not character then return end
	
	local hrp = character:FindFirstChild("HumanoidRootPart")
	if not hrp then return end
	
	local playerPos = hrp.Position
	local counts = {}
	for name in pairs(rarities) do counts[name] = 0 end
	
	local foundOres = {}
	local totalOres = 0
	
	-- Scan all descendants
	for _, obj in ipairs(workspace:GetDescendants()) do
		if not isScanning then break end
		if totalOres >= maxOresLimit then break end
		
		if obj:IsA("ParticleEmitter") then
			for rarityName, data in pairs(rarities) do
				if totalOres >= maxOresLimit then break end
				
				if colorMatches(obj.Color, data.Color) then
					local parent = obj.Parent
					if not parent or not parent:IsA("BasePart") then continue end
					
					-- Check for duplicate
					local isDuplicate = false
					for _, existing in ipairs(foundOres) do
						if existing.Part == parent then
							isDuplicate = true
							break
						end
					end
					if isDuplicate then continue end
					
					local distance = (parent.Position - playerPos).Magnitude
					
					table.insert(foundOres, {
						Part = parent,
						Color = data.Color,
						Rarity = rarityName,
						Distance = distance
					})
					
					counts[rarityName] = (counts[rarityName] or 0) + 1
					totalOres = totalOres + 1
					break -- Only count once per emitter
				end
			end
		end
	end
	
	-- Create visuals for each found ore
	for _, oreInfo in ipairs(foundOres) do
		local oreData = {
			Part = oreInfo.Part,
			Color = oreInfo.Color,
			Rarity = oreInfo.Rarity,
			InitialDistance = oreInfo.Distance
		}
		
		-- Create Highlight (outline)
		local highlight = Instance.new("Highlight")
		highlight.Adornee = oreInfo.Part
		highlight.FillColor = oreInfo.Color
		highlight.OutlineColor = Color3.new(1, 1, 1)
		highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
		highlight.FillTransparency = 0.5
		highlight.OutlineTransparency = 0
		highlight.Parent = oreInfo.Part
		oreData.Highlight = highlight
		
		-- Create Billboard
		local billboard = Instance.new("BillboardGui")
		billboard.Size = UDim2.new(0, 160, 0, showDistance and 50 or 30)
		billboard.StudsOffset = Vector3.new(0, showDistance and 4 or 2.5, 0)
		billboard.AlwaysOnTop = true
		billboard.Parent = oreInfo.Part
		
		-- Rarity label
		if showRarityText then
			local rarityLabel = Instance.new("TextLabel")
			rarityLabel.Name = "RarityLabel"
			rarityLabel.Size = UDim2.fromScale(1, showDistance and 0.6 or 1)
			rarityLabel.BackgroundTransparency = 1
			rarityLabel.TextScaled = true
			rarityLabel.Font = Enum.Font.GothamBold
			rarityLabel.Text = oreInfo.Rarity
			rarityLabel.TextColor3 = oreInfo.Color
			rarityLabel.Parent = billboard
			oreData.RarityLabel = rarityLabel
		end
		
		-- Distance label (will be updated)
		if showDistance then
			local distLabel = Instance.new("TextLabel")
			distLabel.Name = "DistanceLabel"
			distLabel.Size = UDim2.fromScale(1, 0.4)
			distLabel.Position = UDim2.fromScale(0, 0.6)
			distLabel.BackgroundTransparency = 1
			distLabel.TextScaled = true
			distLabel.Font = Enum.Font.Gotham
			distLabel.Text = string.format("%.0f studs", oreInfo.Distance)
			distLabel.TextColor3 = Color3.new(1, 1, 1)
			distLabel.Parent = billboard
			oreData.DistanceLabel = distLabel
		end
		
		oreData.Billboard = billboard
		table.insert(trackedOres, oreData)
	end
	
	-- Update stats display
	local statsText = ""
	for name, count in pairs(counts) do
		if count > 0 then
			statsText = statsText .. string.format("%s: %d\n", name, count)
		end
	end
	statsList.Text = statsText ~= "" and statsText or "No ores found..."
	
	-- Update radar
	updateRadar(foundOres)
	
	-- Start live updates
	startLiveUpdates()
end

-- Update radar dots
function updateRadar(ores)
	-- Clear old dots
	for _, dot in ipairs(radarDots) do
		if dot and dot.Parent then
			dot:Destroy()
		end
	end
	table.clear(radarDots)
	
	local character = player.Character
	if not character then return end
	local hrp = character:FindFirstChild("HumanoidRootPart")
	if not hrp then return end
	
	local playerPos = hrp.Position
	local radarRange = 500
	
	for _, oreData in ipairs(ores) do
		if not oreData.Part or not oreData.Part.Parent then continue end
		
		local diff = oreData.Part.Position - playerPos
		local distance = diff.Magnitude
		
		if distance > radarRange then continue end
		
		-- Calculate relative position
		local relativeX = diff.X / radarRange
		local relativeZ = diff.Z / radarRange
		
		-- Clamp to circle
		local distFromCenter = math.sqrt(relativeX^2 + relativeZ^2)
		if distFromCenter > 1 then
			relativeX = relativeX / distFromCenter
			relativeZ = relativeZ / distFromCenter
		end
		
		-- Create dot
		local dot = Instance.new("Frame")
		dot.Name = "RadarDot"
		dot.Size = UDim2.fromOffset(8, 8)
		dot.Position = UDim2.fromScale(0.5 + relativeX * 0.45, 0.5 + relativeZ * 0.45)
		dot.AnchorPoint = Vector2.new(0.5, 0.5)
		dot.BackgroundColor3 = oreData.Color
		dot.BorderSizePixel = 0
		dot.ZIndex = 2
		
		local dotCorner = Instance.new("UICorner")
		dotCorner.CornerRadius = UDim.new(1, 0)
		dotCorner.Parent = dot
		
		dot.Parent = radarFrame
		table.insert(radarDots, dot)
	end
end
