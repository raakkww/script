-- ========== DYNAMIC REMOTE FINDER ==========

local NodeRemote = nil
local REMOTE_CHECK_INTERVAL = 30

local function findRemotes()
    if Services then
        local clamSvc = Services:FindFirstChild("ClamService")
        if clamSvc then
            local rf = clamSvc:FindFirstChild("RF")
            if rf then
                HatchRemote = rf:FindFirstChild("Purchase") or rf:FindFirstChild("Hatch")
            end
        end
        local nodeSvc = Services:FindFirstChild("NodeService")
        if nodeSvc then
            local re = nodeSvc:FindFirstChild("RE")
            if re then
                NodeRemote = re:FindFirstChild("NodeClicked")
            end
        end
    end
    if not HatchRemote then
        for _, obj in ipairs(ReplicatedStorage:GetDescendants()) do
            if obj:IsA("RemoteFunction") and (obj.Name == "Purchase" or obj.Name == "Hatch") then
                HatchRemote = obj
                break
            end
        end
    end
    if not NodeRemote then
        for _, obj in ipairs(ReplicatedStorage:GetDescendants()) do
            if obj:IsA("RemoteEvent") and obj.Name == "NodeClicked" then
                NodeRemote = obj
                break
            end
        end
    end
end
findRemotes()

task.spawn(function()
    while true do
        task.wait(REMOTE_CHECK_INTERVAL)
        findRemotes()
    end
end)

-- ========== EGG LIST ==========
local function generateEggs()
    local eggs = {}
    local added = {}
    for _, i in ipairs(validAreaNumbers) do
        local basic = "area" .. i .. " basic"
        if not added[basic] then
            added[basic] = true
            table.insert(eggs, basic)
        end
        local golden = "area" .. i .. " golden"
        if not added[golden] then
            added[golden] = true
            table.insert(eggs, golden)
        end
    end
    table.insert(eggs, "meme buddies")
    table.sort(eggs)
    return eggs
end
local availableEggs = generateEggs()

-- ========== AUTO ATTACK ==========
task.spawn(function()
    while true do
        if State.autoAttack and NodeRemote then
            pcall(function()
                local char = player.Character
                local hrp = char and char:FindFirstChild("HumanoidRootPart")
                if not hrp then return end
                local nodes = Workspace:FindFirstChild("Nodes")
                if nodes then
                    local closest, minDist = nil, 150
                    for _, node in pairs(nodes:GetChildren()) do
                        if node:IsA("BasePart") then
                            local dist = (node.Position - hrp.Position).Magnitude
                            if dist < minDist then
                                minDist = dist
                                closest = node
                            end
                        end
                    end
                    if closest then
                        NodeRemote:FireServer(closest, true, false)
                    end
                end
            end)
        end
        task.wait(0.2)
    end
end)

-- ========== AUTO BOSS PARTS ==========
local bossSpawnPositions = {}
local function initBossSpawnPositions()
    local prog = Workspace:FindFirstChild("Programmables")
    if not prog then return end
    local spawners = prog:FindFirstChild("BossPartSpawners")
    if not spawners then return end
    for _, worldName in ipairs({"GooLagoon", "Mermalair", "VacationWorld"}) do
        local folder = spawners:FindFirstChild(worldName)
        if folder then
            for _, obj in ipairs(folder:GetDescendants()) do
                if obj:IsA("BasePart") or obj:IsA("MeshPart") then
                    table.insert(bossSpawnPositions, obj.Position)
                    break
                end
            end
        end
    end
end
initBossSpawnPositions()

task.spawn(function()
    local collected = {}
    local lastPosition = nil
    local teleporting = false
    while true do
        if State.autoBossParts and not teleporting then
            pcall(function()
                local char = player.Character
                local hrp = char and char:FindFirstChild("HumanoidRootPart")
                if not hrp then return end
                local terrain = Workspace:FindFirstChild("Terrain")
                if terrain then
                    for _, obj in pairs(terrain:GetChildren()) do
                        if obj:IsA("Attachment") and obj.Name:match("^Item") and not collected[obj] then
                            local itemPos = obj.WorldPosition
                            local isBossPart = false
                            for _, spawnPos in ipairs(bossSpawnPositions) do
                                if (itemPos - spawnPos).Magnitude < 120 then
                                    isBossPart = true
                                    break
                                end
                            end
                            if isBossPart then
                                collected[obj] = true
                                teleporting = true
                                lastPosition = hrp.CFrame
                                hrp.CFrame = CFrame.new(itemPos + Vector3.new(0, 3, 0))
                                task.wait(0.3)
                                if lastPosition then
                                    hrp.CFrame = lastPosition
                                end
                                teleporting = false
                                task.delay(15, function() collected[obj] = nil end)
                                break
                            else
                                collected[obj] = true
                                task.delay(5, function() collected[obj] = nil end)
                            end
                        end
                    end
                end
            end)
        end
        task.wait(0.1)
    end
end)

-- ========== AUTO PEDESTALS ==========
local function findPedestals()
    local items = {}
    local prog = Workspace:FindFirstChild("Programmables")
    if prog then
        local pedestalsFolder = prog:FindFirstChild("CollectiblePedestals") or prog
        for _, obj in pairs(pedestalsFolder:GetDescendants()) do
            if obj:IsA("BasePart") and (obj.Name == "CollectiblePedestal" or (obj.Parent and obj.Parent.Name == "CollectiblePedestal")) then
                table.insert(items, obj)
            end
        end
    end
    return items
end

local pedestalCache = {}
local function startPedestals()
    task.spawn(function()
        while State.autoPedestals do
            local pedestals = findPedestals()
            local found = false
            for _, ped in ipairs(pedestals) do
                if not State.autoPedestals then break end
                if ped and ped.Parent and not pedestalCache[ped] then
                    pedestalCache[ped] = true
                    found = true
                    local char = player.Character
                    local hrp = char and char:FindFirstChild("HumanoidRootPart")
                    if hrp then
                        hrp.CFrame = CFrame.new(ped.Position + Vector3.new(0, 5, 0))
                        local cd = ped:FindFirstChildOfClass("ClickDetector")
                        if cd then fireclickdetector(cd) end
                        local pp = ped:FindFirstChildOfClass("ProximityPrompt")
                        if pp then fireproximityprompt(pp) end
                        task.wait(0.3)
                        task.delay(10, function() pedestalCache[ped] = nil end)
                    end
                end
            end
            if found then
                task.wait(0.5)
            else
                task.wait(2)
            end
        end
    end)
end

-- ========== AUTO RACE WITH FAST TWEEN ==========
local function getCheckpointsOrdered()
    local checkpoints = Workspace:FindFirstChild("Checkpoints")
    if not checkpoints then
        local prog = Workspace:FindFirstChild("Programmables")
        if prog then
            local races = prog:FindFirstChild("Races")
            if races then
                local boatSchool = races:FindFirstChild("boat school")
                if boatSchool then
                    checkpoints = boatSchool:FindFirstChild("Checkpoints")
                end
            end
        end
    end
    if not checkpoints then return {} end

    local cpMap = {}
    for _, child in ipairs(checkpoints:GetChildren()) do
        if child:IsA("BasePart") then
            local num = tonumber(child.Name)
            if num then
                cpMap[num] = child
            end
        end
    end
    local ordered = {}
    for i = 1, #cpMap do
        if cpMap[i] then
            table.insert(ordered, cpMap[i])
        else
            break
        end
    end
    return ordered
end

local function getPlayerVehicle()
    local char = player.Character
    if not char then return nil end
    local humanoid = char:FindFirstChildOfClass("Humanoid")
    if not humanoid then return nil end
    for _, v in ipairs(Workspace:GetDescendants()) do
        if v:IsA("VehicleSeat") and v.Occupant == humanoid then
            return v.Parent
        end
    end
    return nil
end

local function smoothMoveTo(targetCF, vehicle)
    local char = player.Character
    if not char then return end
    local part = vehicle and vehicle.PrimaryPart or char:FindFirstChild("HumanoidRootPart")
    if not part then return end
    -- Very fast tween (0.1s) to trigger the checkpoint
    local tween = TweenService:Create(part, TweenInfo.new(0.1, Enum.EasingStyle.Linear), {CFrame = targetCF})
    tween:Play()
    tween.Completed:Wait()
end

local function triggerTouch(part, checkpoint)
    if not part or not checkpoint then return end
    firetouchinterest(part, checkpoint, 0)
    task.wait(0.05)
    firetouchinterest(part, checkpoint, 1)
end

local function startRace()
    task.spawn(function()
        while State.autoRace do
            local checkpoints = getCheckpointsOrdered()
            if #checkpoints > 0 then
                for _, cp in ipairs(checkpoints) do
                    if not State.autoRace then break end

                    local vehicle = getPlayerVehicle()
                    local char = player.Character
                    local hrp = char and char:FindFirstChild("HumanoidRootPart")
                    if not hrp then continue end

                    local targetCF = cp.CFrame + Vector3.new(0, 3, 0)

                    -- Smooth fast movement
                    smoothMoveTo(targetCF, vehicle)

                    -- Trigger touch
                    if vehicle then
                        local touchPart = vehicle.PrimaryPart or vehicle:FindFirstChildWhichIsA("BasePart")
                        if touchPart then
                            triggerTouch(touchPart, cp)
                        end
                    else
                        triggerTouch(hrp, cp)
                    end

                    task.wait(0.1)
                end
                task.wait(0.5)
            else
                task.wait(2)
            end
        end
    end)
end

-- ========== AUTO HATCH WITH ANIMATION SKIP (destroys GUI) ==========
local hatchSession = 0
local HATCH_COOLDOWN = 3 -- seconds between hatches

local function findIncubator()
    local incubatorFolder = Workspace:FindFirstChild("Incubators") or Workspace:FindFirstChild("EggMachines")
    if incubatorFolder then
        for _, obj in ipairs(incubatorFolder:GetChildren()) do
            if obj:IsA("BasePart") then
                return obj
            end
        end
    end
    for _, obj in ipairs(Workspace:GetDescendants()) do
        if obj:IsA("ClickDetector") or obj:IsA("ProximityPrompt") then
            local parent = obj.Parent
            if parent and parent:IsA("BasePart") and parent.Name:lower():find("hatch") then
                return parent
            end
        end
    end
    return nil
end

local function startHatch()
    if not selectedEgg or not HatchRemote then
        warn("Hatch not available: egg not selected or remote missing")
        return
    end
    hatchSession = hatchSession + 1
    local currentSession = hatchSession

    task.spawn(function()
        while State.autoHatch and currentSession == hatchSession do
            -- Move near an incubator
            local incubator = findIncubator()
            if incubator then
                local char = player.Character
                local hrp = char and char:FindFirstChild("HumanoidRootPart")
                if hrp then
                    hrp.CFrame = incubator.CFrame * CFrame.new(0, 3, 0)
                    task.wait(0.2)
                end
            else
                warn("No incubator found, proceeding without moving")
            end

            -- Invoke remote
            local success = pcall(function()
                HatchRemote:InvokeServer(selectedEgg, hatchAmount)
            end)

            if success then
                -- DESTROY ANIMATION IMMEDIATELY (if present)
                local gui = player:FindFirstChildOfClass("PlayerGui")
                if gui then
                    local anim = gui:FindFirstChild("HatchAnimation")
                    if anim then
                        anim:Destroy()
                    end
                end
                -- Short pause for safety (server already got the request)
                task.wait(0.1)
            else
                warn("Hatch call failed, retrying soon")
                task.wait(1)
            end

            -- Cooldown before next hatch
            task.wait(HATCH_COOLDOWN)
        end
    end)
end

local function stopHatch()
    State.autoHatch = false
    hatchSession = hatchSession + 1
end
