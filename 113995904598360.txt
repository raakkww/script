local function Notify(text, duration)
    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(1, 0, 0, 40)
    frame.BackgroundColor3 = THEME.Background
    frame.BackgroundTransparency = 0.1
    frame.Position = UDim2.new(1, 10, 0, 0)
    
    local corner = Instance.new("UICorner", frame)
    corner.CornerRadius = UDim.new(0, 6)
    
    local stroke = Instance.new("UIStroke", frame)
    stroke.Color = THEME.Accent
    stroke.Thickness = 1
    
    local label = Instance.new("TextLabel", frame)
    label.Size = UDim2.new(1, -10, 1, 0)
    label.Position = UDim2.new(0, 10, 0, 0)
    label.BackgroundTransparency = 1
    label.Text = text
    label.TextColor3 = THEME.Text
    label.Font = Enum.Font.GothamSemibold
    label.TextSize = 12
    label.TextXAlignment = Enum.TextXAlignment.Left
    
    frame.Parent = notifyContainer
    Tween(frame, {0.5, Enum.EasingStyle.Back, Enum.EasingDirection.Out}, {Position = UDim2.new(0, 0, 0, 0)})
    
    task.delay(duration or 3, function()
        if frame and frame.Parent then
            Tween(frame, {0.5, Enum.EasingStyle.Back, Enum.EasingDirection.In}, {Position = UDim2.new(1, 10, 0, 0)})
            task.wait(0.5)
            frame:Destroy()
        end
    end)
end

-- Lógica Funcional
local selectedBase, selectedChar = nil, nil

local function getBases()
    local b = {}
    if workspace:FindFirstChild("Bases") then
        for _, base in pairs(workspace.Bases:GetChildren()) do
            local s = base:FindFirstChild("Sign")
            if s and s:FindFirstChild("SignPart") and s.SignPart:FindFirstChild("SurfaceGui") then
                local t = s.SignPart.SurfaceGui:FindFirstChild("TextLabel")
                if t and t.Text then
                    local name = string.match(t.Text, "(.+)'s Base")
                    if name then table.insert(b, {Obj = base, Name = name}) end
                end
            end
        end
    end
    return b
end

local function getChars(base)
    local c = {}
    local folders = {"Characters", "RainbowCharacters", "CosmicCharacters"}
    for _, fName in ipairs(folders) do
        local f = base:FindFirstChild(fName)
        if f then
            for _, char in pairs(f:GetChildren()) do
                if char:IsA("Model") then table.insert(c, {Obj = char, Name = char.Name, Folder = fName}) end
            end
        end
    end
    return c
end

local function updateBases()
    for _, v in pairs(baseScroll:GetChildren()) do if v:IsA("TextButton") then v:Destroy() end end
    for _, data in ipairs(getBases()) do
        local b = Instance.new("TextButton", baseScroll)
        b.Size = UDim2.new(1, 0, 0, 30)
        b.BackgroundTransparency = 1
        b.Text = "  " .. data.Name
        b.TextColor3 = THEME.Text
        b.TextXAlignment = Enum.TextXAlignment.Left
        b.Font = Enum.Font.Gotham
        b.TextSize = 12
        b.ZIndex = 11
        
        b.MouseEnter:Connect(function() b.BackgroundTransparency = 0.8; b.BackgroundColor3 = THEME.Accent end)
        b.MouseLeave:Connect(function() b.BackgroundTransparency = 1 end)
        
        b.MouseButton1Click:Connect(function()
            selectedBase = data.Obj
            baseBtn.Text = data.Name
            baseBtn.TextColor3 = THEME.Accent
            baseScroll.Visible = false
            Notify("Base de " .. data.Name .. " seleccionada")
            updateChars()
        end)
    end
    baseScroll.CanvasSize = UDim2.new(0, 0, 0, #baseScroll:GetChildren() * 32)
end

function updateChars()
    for _, v in pairs(charScroll:GetChildren()) do if v:IsA("TextButton") then v:Destroy() end end
    if not selectedBase then return end
    for _, data in ipairs(getChars(selectedBase)) do
        local b = Instance.new("TextButton", charScroll)
        b.Size = UDim2.new(1, 0, 0, 30)
        b.BackgroundTransparency = 1
        b.Text = "  " .. data.Name .. " [" .. data.Folder .. "]"
        b.TextColor3 = THEME.Text
        b.TextXAlignment = Enum.TextXAlignment.Left
        b.Font = Enum.Font.Gotham
        b.TextSize = 12
        b.ZIndex = 11
        
        b.MouseEnter:Connect(function() b.BackgroundTransparency = 0.8; b.BackgroundColor3 = THEME.Accent end)
        b.MouseLeave:Connect(function() b.BackgroundTransparency = 1 end)
        
        b.MouseButton1Click:Connect(function()
            selectedChar = data.Obj
            charBtn.Text = data.Name
            charBtn.TextColor3 = THEME.Accent
            charScroll.Visible = false
            Notify("Personaje " .. data.Name .. " listo")
        end)
    end
    charScroll.CanvasSize = UDim2.new(0, 0, 0, #charScroll:GetChildren() * 32)
end

baseBtn.MouseButton1Click:Connect(function() baseScroll.Visible = not baseScroll.Visible; charScroll.Visible = false; updateBases() end)
charBtn.MouseButton1Click:Connect(function() if selectedBase then charScroll.Visible = not charScroll.Visible; baseScroll.Visible = false; updateChars() else Notify("¡Selecciona una base primero!", 2) end end)

-- Ejecución de Robo
stealBtn.MouseButton1Click:Connect(function()
    if not selectedBase or not selectedChar then return Notify("Faltan datos de selección", 2) end
    
    local char = localPlayer.Character or localPlayer.CharacterAdded:Wait()
    local hrp = char:WaitForChild("HumanoidRootPart")
    local hum = char:WaitForChild("Humanoid")
    local targetHrp = selectedChar:FindFirstChild("HumanoidRootPart")
    
    if not targetHrp then return Notify("Error: RootPart no encontrado") end
    
    Notify("Iniciando secuencia de robo...")
    progressBg.Visible = true
    progressBar.Size = UDim2.new(0, 0, 1, 0)
    
    -- Noclip & Fly
    local noclip = RunService.Stepped:Connect(function()
        if char then
            for _, p in pairs(char:GetDescendants()) do if p:IsA("BasePart") then p.CanCollide = false end end
        end
    end)
    hum.PlatformStand = true
    
    hrp.CFrame = targetHrp.CFrame * CFrame.new(0, 3, 6)
    
    local bg = Instance.new("BodyGyro", hrp)
    bg.MaxTorque = Vector3.new(4e5, 4e5, 4e5)
    local bv = Instance.new("BodyVelocity", hrp)
    bv.MaxForce = Vector3.new(4e5, 4e5, 4e5)
    bv.Velocity = Vector3.zero

    local fly = RunService.Heartbeat:Connect(function()
        if hrp and targetHrp then
            bg.CFrame = CFrame.lookAt(hrp.Position, targetHrp.Position) * CFrame.Angles(math.rad(-90), 0, 0)
        end
    end)

    -- Animación de barra
    Tween(progressBar, {6, Enum.EasingStyle.Linear}, {Size = UDim2.new(1, 0, 1, 0)})
    task.wait(6)

    -- Retorno
    local function getMyBase()
        for _, b in pairs(workspace.Bases:GetChildren()) do
            local s = b:FindFirstChild("Sign")
            if s and s:FindFirstChild("SignPart") then
                local t = s.SignPart.SurfaceGui.TextLabel
                if string.find(t.Text, localPlayer.Name) then return b end
            end
        end
    end

    local myBase = getMyBase()
    if myBase and myBase:FindFirstChild("StealCollect2") then
        hrp.CFrame = myBase.StealCollect2.CFrame + Vector3.new(0, 3, 0)
    end

    -- Limpieza
    if bg then bg:Destroy() end
    if bv then bv:Destroy() end
    if fly then fly:Disconnect() end
    if noclip then noclip:Disconnect() end
    
    if hum then hum.PlatformStand = false end
    if char then
        for _, p in pairs(char:GetDescendants()) do if p:IsA("BasePart") then p.CanCollide = true end end
    end
    
    progressBg.Visible = false
    Notify("¡Robo completado con éxito! ✅")
end)

    task.wait(0.5)
    screenGui:Destroy()
end)

-- Lógica Funcional
local selectedBase, selectedChar = nil, nil

local function getBases()
    local b = {}
    if workspace:FindFirstChild("Bases") then
        for _, base in pairs(workspace.Bases:GetChildren()) do
            local s = base:FindFirstChild("Sign")
            if s and s:FindFirstChild("SignPart") and s.SignPart:FindFirstChild("SurfaceGui") then
                local t = s.SignPart.SurfaceGui:FindFirstChild("TextLabel")
                if t and t.Text then
                    local name = string.match(t.Text, "(.+)'s Base")
                    if name then table.insert(b, {Obj = base, Name = name}) end
                end
            end
        end
    end
    return b
end

local function getChars(base)
    local c = {}
    local folders = {"Characters", "RainbowCharacters", "CosmicCharacters"}
    for _, fName in ipairs(folders) do
        local f = base:FindFirstChild(fName)
        if f then
            for _, char in pairs(f:GetChildren()) do
                if char:IsA("Model") then table.insert(c, {Obj = char, Name = char.Name, Folder = fName}) end
            end
        end
    end
    return c
end

local function updateBases()
    for _, v in pairs(baseScroll:GetChildren()) do if v:IsA("TextButton") then v:Destroy() end end
    for _, data in ipairs(getBases()) do
        local b = Instance.new("TextButton", baseScroll)
        b.Size = UDim2.new(1, 0, 0, 30)
        b.BackgroundTransparency = 1
        b.Text = "  " .. data.Name
        b.TextColor3 = THEME.Text
        b.TextXAlignment = Enum.TextXAlignment.Left
        b.Font = Enum.Font.Gotham
        b.TextSize = 12
        b.ZIndex = 11
        
        b.MouseEnter:Connect(function() b.BackgroundTransparency = 0.8; b.BackgroundColor3 = THEME.Accent end)
        b.MouseLeave:Connect(function() b.BackgroundTransparency = 1 end)
        
        b.MouseButton1Click:Connect(function()
            selectedBase = data.Obj
            baseBtn.Text = data.Name
            baseBtn.TextColor3 = THEME.Accent
            baseScroll.Visible = false
            Notify("Base de " .. data.Name .. " seleccionada")
            updateChars()
        end)
    end
    baseScroll.CanvasSize = UDim2.new(0, 0, 0, #baseScroll:GetChildren() * 32)
end

function updateChars()
    for _, v in pairs(charScroll:GetChildren()) do if v:IsA("TextButton") then v:Destroy() end end
    if not selectedBase then return end
    for _, data in ipairs(getChars(selectedBase)) do
        local b = Instance.new("TextButton", charScroll)
        b.Size = UDim2.new(1, 0, 0, 30)
        b.BackgroundTransparency = 1
        b.Text = "  " .. data.Name .. " [" .. data.Folder .. "]"
        b.TextColor3 = THEME.Text
        b.TextXAlignment = Enum.TextXAlignment.Left
        b.Font = Enum.Font.Gotham
        b.TextSize = 12
        b.ZIndex = 11
        
        b.MouseEnter:Connect(function() b.BackgroundTransparency = 0.8; b.BackgroundColor3 = THEME.Accent end)
        b.MouseLeave:Connect(function() b.BackgroundTransparency = 1 end)
        
        b.MouseButton1Click:Connect(function()
            selectedChar = data.Obj
            charBtn.Text = data.Name
            charBtn.TextColor3 = THEME.Accent
            charScroll.Visible = false
            Notify("Personaje " .. data.Name .. " listo")
        end)
    end
    charScroll.CanvasSize = UDim2.new(0, 0, 0, #charScroll:GetChildren() * 32)
end

baseBtn.MouseButton1Click:Connect(function() baseScroll.Visible = not baseScroll.Visible; charScroll.Visible = false; updateBases() end)
charBtn.MouseButton1Click:Connect(function() if selectedBase then charScroll.Visible = not charScroll.Visible; baseScroll.Visible = false; updateChars() else Notify("¡Selecciona una base primero!", 2) end end)

-- Ejecución de Robo
stealBtn.MouseButton1Click:Connect(function()
    if not selectedBase or not selectedChar then return Notify("Faltan datos de selección", 2) end
    
    local char = localPlayer.Character or localPlayer.CharacterAdded:Wait()
    local hrp = char:WaitForChild("HumanoidRootPart")
    local hum = char:WaitForChild("Humanoid")
    local targetHrp = selectedChar:FindFirstChild("HumanoidRootPart")
    
    if not targetHrp then return Notify("Error: RootPart no encontrado") end
    
    Notify("Iniciando secuencia de robo...")
    progressBg.Visible = true
    progressBar.Size = UDim2.new(0, 0, 1, 0)
    
    -- Noclip & Fly
    local noclip = RunService.Stepped:Connect(function()
        if char then
            for _, p in pairs(char:GetDescendants()) do if p:IsA("BasePart") then p.CanCollide = false end end
        end
    end)
    hum.PlatformStand = true
    
    hrp.CFrame = targetHrp.CFrame * CFrame.new(0, 3, 6)
    
    local bg = Instance.new("BodyGyro", hrp)
    bg.MaxTorque = Vector3.new(4e5, 4e5, 4e5)
    local bv = Instance.new("BodyVelocity", hrp)
    bv.MaxForce = Vector3.new(4e5, 4e5, 4e5)
    bv.Velocity = Vector3.zero

    local fly = RunService.Heartbeat:Connect(function()
        if hrp and targetHrp then
            bg.CFrame = CFrame.lookAt(hrp.Position, targetHrp.Position) * CFrame.Angles(math.rad(-90), 0, 0)
        end
    end)

    -- Animación de barra
    Tween(progressBar, {6, Enum.EasingStyle.Linear}, {Size = UDim2.new(1, 0, 1, 0)})
    task.wait(6)

    -- Retorno
    local function getMyBase()
        for _, b in pairs(workspace.Bases:GetChildren()) do
            local s = b:FindFirstChild("Sign")
            if s and s:FindFirstChild("SignPart") then
                local t = s.SignPart.SurfaceGui.TextLabel
                if string.find(t.Text, localPlayer.Name) then return b end
            end
        end
    end

    local myBase = getMyBase()
    if myBase and myBase:FindFirstChild("StealCollect2") then
        hrp.CFrame = myBase.StealCollect2.CFrame + Vector3.new(0, 3, 0)
    end

    -- Limpieza
    if bg then bg:Destroy() end
    if bv then bv:Destroy() end
    if fly then fly:Disconnect() end
    if noclip then noclip:Disconnect() end
    
    if hum then hum.PlatformStand = false end
    if char then
        for _, p in pairs(char:GetDescendants()) do if p:IsA("BasePart") then p.CanCollide = true end end
    end
    
    progressBg.Visible = false
    Notify("Robbery successfully completed! ✅")
end)
