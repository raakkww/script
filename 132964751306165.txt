} 
local AntiSection = CustomsTab:NewSection("Custom Anticheat")
AntiSection:NewButton("Disable Custom Anticheat (Perm)", "Disables all anticheats at once", function()
    for _, name in ipairs(customNames) do
        local args = {name, 0}
        game:GetService("ReplicatedStorage").CoinsChecknRemove:FireServer(unpack(args))
    end
end)

local MorphSection = CustomsTab:NewSection("Custom Morpher")

for _, name in ipairs(customNames) do
    local buttonName = (name == "HECATE GODDESS OF WITCHCRAFT" and "Hecate") or name
    MorphSection:NewButton(buttonName, "Morph into " .. name, function()
        local args = {"Outfit1", name}
        game:GetService("ReplicatedStorage").System_Remotes.CharacterChoose:FireServer(unpack(args))
    end)
end
