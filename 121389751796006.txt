getgenv().EndRescueLoop = false

   Name = "",
   LoadingTitle = "",
   LoadingSubtitle = "",
   ConfigurationSaving = {
      Enabled = true,
      FolderName = "SandManHub",
      FileName = "v2.2"
   },

   },

local MainTab = Window:CreateTab("Era Of Supes", nil)

-- EndCrime Spam Toggle (SLOWER: 0.5s delay)
MainTab:CreateToggle({
   Name = "Inf Money And Levels",
   CurrentValue = false,
   Flag = "Inf STUFF",
   Callback = function(Value)
      getgenv().EndRescueLoop = Value
      if Value then
         task.spawn(function()
            while getgenv().EndRescueLoop do
               local args = { "EndCrime", {} }
               pcall(function()
                  game:GetService("ReplicatedStorage"):WaitForChild("Events"):WaitForChild("CrimeHandler"):FireServer(unpack(args))
               end)
               task.wait(0.5)  -- Slower loop (2x/sec max)
            end
         end)
         Rayfield:Notify({
            Title = "Inf ACTIVE",
            Content = "Spamming EndCrime (slower loop)... (Toggle off to stop)",
            Duration = 2.5,
         })
      else
         Rayfield:Notify({
            Title = "Inf STOPPED",
            Content = "Loop disconnected safely.",
            Duration = 2,
         })
      end
   end,
})
