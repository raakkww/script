local buttonCorner = Instance.new("UICorner")
buttonCorner.CornerRadius = UDim.new(0, 6)
buttonCorner.Parent = afkButton

local afkEnabled = false
local farmConnection = nil
local targetPos = Vector3.new(9201.10, -16.51 + 5, -59.23)

local function toggleAFK()
    afkEnabled = not afkEnabled
    
    if afkEnabled then
        statusLabel.Text = "Status: ON"
        statusLabel.TextColor3 = Color3.fromRGB(0, 255, 0)
        afkButton.Text = "AFK FARM: ON"
        afkButton.BackgroundColor3 = Color3.fromRGB(0, 150, 0)
        
        local character = player.Character or player.CharacterAdded:Wait()
        local humanoidRootPart = character:WaitForChild("HumanoidRootPart")
        humanoidRootPart.CFrame = CFrame.new(targetPos)
        
        farmConnection = RunService.Heartbeat:Connect(function()
            if character and character.Parent and humanoidRootPart and humanoidRootPart.Parent then
                local offset = Vector3.new(0, math.sin(tick() * 5) * 0.5, 0)
                humanoidRootPart.CFrame = CFrame.new(targetPos + offset)
            else
                if farmConnection then farmConnection:Disconnect() end
            end
        end)
    else
        statusLabel.Text = "Status: CLOSE"
        statusLabel.TextColor3 = Color3.fromRGB(255, 0, 0)
        afkButton.Text = "AFK FARM: OFF"
        afkButton.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
        
        if farmConnection then
            farmConnection:Disconnect()
            farmConnection = nil
        end
    end
end

afkButton.MouseButton1Click:Connect(toggleAFK)

-- // SÜRÜKLENEBİLİR YAPMA (DRAGGING SYSTEM)
local dragging, dragInput, dragStart, startPos

frame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = frame.Position
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement then
        dragInput = input
    end
end)

UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = false
    end
end)

RunService.RenderStepped:Connect(function()
    if dragging and dragInput then
        local delta = dragInput.Position - dragStart
        frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)

-- // INSERT KEY TOGGLE
local guiVisible = true
UserInputService.InputBegan:Connect(function(input)
    if input.KeyCode == Enum.KeyCode.Insert then
        guiVisible = not guiVisible
        frame.Visible = guiVisible
    end
end)
