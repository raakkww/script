-- // caching
local MINE_LOCATIONS = {"Mine", "Mine2", "Mine3", "SpaceMine"}
local cachedPickaxeRF = nil
local cachedGodmodeRF = nil
local currentRF = nil

-- // functions
local function findRemoteFunction(parent, namePattern)
    if not parent then return nil end
    for _, item in ipairs(parent:GetChildren()) do
        if item:IsA("Tool") then
            local name = string_lower(item.Name)
            if not namePattern or name:find(namePattern) then
                local rf = item:FindFirstChildOfClass("RemoteFunction")
                if rf then return rf end
            end
        end
    end
    return nil
end

local function findRemoteInContainer(container, namePattern)
    if not container then return nil end
    for _, item in ipairs(container:GetChildren()) do
        if item:IsA("Tool") then
            local name = string_lower(item.Name)
            if not namePattern or name:find(namePattern) then
                local rf = item:FindFirstChildOfClass("RemoteFunction")
                if rf then return rf end
            end
        end
    end
    return nil
end

local function GetGodmodeEvent()
    if cachedGodmodeRF and cachedGodmodeRF.Parent then
        return cachedGodmodeRF
    end
    
    local character = LocalPlayer.Character
    if not character then return nil end

    -- check character and backpack for armour
    for _, container in ipairs { character, LocalPlayer.Backpack } do
        local rf = findRemoteFunction(container, "armour")
        if rf then cachedGodmodeRF = rf; return rf end
    end

    -- check workspace models for armour
    for _, model in ipairs(workspace:GetChildren()) do
        if model:IsA("Model") then
            local rf = findRemoteFunction(model, "armour")
            if rf then cachedGodmodeRF = rf; return rf end
        end
    end

    return nil
end

local function ApplyGodmode()
    local Event = GetGodmodeEvent()
    if not Event then return end

    local function ProtectTarget(targetModel)
        if not targetModel then return end
        local humanoid = targetModel:FindFirstChildOfClass("Humanoid")
        if humanoid then
            pcall(function()
                Event:InvokeServer("protect", {targetModel, humanoid, G.godmodeHealth or 99999999})
            end)
        end
    end

    if G.godmodeTarget == "self" then
        ProtectTarget(LocalPlayer.Character)
    elseif G.godmodeTarget == "all players" then
        for _, player in ipairs(Players:GetPlayers()) do
            if player.Character then
                ProtectTarget(player.Character)
            end
        end
    elseif G.godmodeTarget == "specific player" and G.godmodeTargetPlayer then
        local targetPlayer = Players:FindFirstChild(G.godmodeTargetPlayer)
        if targetPlayer and targetPlayer.Character then
            ProtectTarget(targetPlayer.Character)
        end
    end
end

local function GodmodeLoop()
    while G.godmodeEnabled do
        ApplyGodmode()
        task.wait(G.godmodeInterval or 1)
    end
end

local function enableNoCooldown()
    local old
    old = hookfunction(wait, newcclosure(function(...)
        return old(0)
    end))
end

local function getPickaxeTool()
    if cachedPickaxeRF and cachedPickaxeRF.Parent then
        return cachedPickaxeRF
    end

    local character = LocalPlayer.Character
    if not character then return nil end

    -- manual: get any equipped tool
    if autoMineTool == "manual (equipped tool)" then
        for _, item in ipairs(character:GetChildren()) do
            if item:IsA("Tool") then
                local rf = item:FindFirstChildOfClass("RemoteFunction")
                if rf then cachedPickaxeRF = rf; return rf end
            end
        end
    end

    -- auto: search for tools with "pickaxe" in the name in character and backpack
    for _, container in ipairs { character, LocalPlayer.Backpack } do
        local rf = findRemoteFunction(container, "pickaxe")
        if rf then cachedPickaxeRF = rf; return rf end
    end

    return nil
end

local function getAvailableOres()
    local ores = {}
    for _, mineName in ipairs(MINE_LOCATIONS) do
        if G.autoMineLocation == "all" or G.autoMineLocation == mineName then
            local mine = workspace:FindFirstChild(mineName)
            if mine then
                for _, oreType in ipairs(mine:GetChildren()) do
                    if oreType:IsA("Model") or oreType:IsA("Folder") then
                        for _, ore in ipairs(oreType:GetChildren()) do
                            if string_lower(ore.Name) == "ore" then
                                table_insert(ores, mineName .. "." .. oreType.Name .. "." .. ore.Name)
                            end
                        end
                    end
                end
            end
        end
    end
    return ores
end

local function getOreFromPath(orePath)
    if not orePath or orePath == "" then return nil end
    local parts = string.split(orePath, ".")
    if #parts < 3 then return nil end

    local mine = workspace:FindFirstChild(parts[1])
    if not mine then return nil end
    local oreFolder = mine:FindFirstChild(parts[2])
    if not oreFolder then return nil end
    return oreFolder:FindFirstChild(parts[3])
end

local function mineAllOres()
    local rf = getPickaxeTool()
    if not rf then return end

    for _, mineName in ipairs(MINE_LOCATIONS) do
        local mine = workspace:FindFirstChild(mineName)
        if mine then
            for _, oreType in ipairs(mine:GetChildren()) do
                if oreType:IsA("Model") or oreType:IsA("Folder") then
                    for _, ore in ipairs(oreType:GetChildren()) do
                        if string_lower(ore.Name) == "ore" then
                            pcall(function() rf:InvokeServer("hit", {ore, G.autoMineDamage}) end)
                        end
                    end
                end
            end
        end
    end
end

local function AutoMineLoop()
    local rf = getPickaxeTool()
    if not rf then G.autoMineLoopRunning = false; return end

    while G.autoMineEnabled and rf and rf.Parent do
        for _, mineName in ipairs(MINE_LOCATIONS) do
            local mine = workspace:FindFirstChild(mineName)
            if mine then
                for _, oreType in ipairs(mine:GetChildren()) do
                    if oreType:IsA("Model") or oreType:IsA("Folder") then
                        for _, ore in ipairs(oreType:GetChildren()) do
                            if string_lower(ore.Name) == "ore" then
                                pcall(function() rf:InvokeServer("hit", {ore, G.autoMineDamage or 99999999}) end)
                            end
                        end
                    end
                end
            end
        end
        task.wait(G.autoMineInterval or 0.5)
    end
    G.autoMineLoopRunning = false
end

local function getToolRemote()
    if currentRF and currentRF.Parent then
        return currentRF
    end
    currentRF = nil

    for _, container in ipairs { LocalPlayer.Backpack, LocalPlayer.Character } do
        if not container then continue end
        for _, item in ipairs(container:GetChildren()) do
            if item:IsA("Tool") then
                local rf = item:FindFirstChildOfClass("RemoteFunction")
                if rf then currentRF = rf; return rf end
            end
        end
    end
end

local function getAliveEnemies()
    local enemies = {}
    for _, model in ipairs(workspace:GetChildren()) do
        if not model:IsA("Model") then continue end
        local head = model:FindFirstChild("Head")
        if not head then continue end
        local humanoid = model:FindFirstChildOfClass("Humanoid")
        if not humanoid or humanoid.Health <= 0 then continue end
        if model:FindFirstChild("EnemyMain") then
            table_insert(enemies, head)
        end
    end
    return enemies
end

local function scanTool(tool)
    if not tool:IsA("Tool") then return end
    local rf = tool:FindFirstChildOfClass("RemoteFunction")
    if rf then currentRF = rf end
end

local function initCharacter(char)
    currentRF = nil
    for _, container in ipairs { char, LocalPlayer.Backpack } do
        if not container then continue end
        for _, item in ipairs(container:GetChildren()) do
            scanTool(item)
        end
    end
    char.ChildAdded:Connect(scanTool)
end

local function init()
    LocalPlayer.Backpack.ChildAdded:Connect(scanTool)
    if LocalPlayer.Character then
        initCharacter(LocalPlayer.Character)
    end
    LocalPlayer.CharacterAdded:Connect(initCharacter)
    currentRF = nil
    for _, container in ipairs { LocalPlayer.Character, LocalPlayer.Backpack } do
        if not container then continue end
        for _, item in ipairs(container:GetChildren()) do
            scanTool(item)
        end
    end
end

local function MeteorLoop()
    while G.enabled do
        local rf = getToolRemote()
        if rf then
            for _, head in ipairs(getAliveEnemies()) do
                pcall(rf.InvokeServer, rf, "meteor", { math.huge, head })
            end
        end
        RunService.Heartbeat:wait()
    end
    G.meteorLoopRunning = false
end

local function Hook_Adonis(metadefs)
    for _, tbl in metadefs do
        for _, func in tbl do
            if type(func) == "function" and islclosure(func) then
                hookfunction(func, function() return pcall(coroutine.close, coroutine.running()) end)
            end
        end
    end
end

local function Init_Bypass()
    for _, v in getgc(true) do
        if typeof(v) == "table"
            and rawget(v, "indexInstance")
            and rawget(v, "newindexInstance")
            and rawget(v, "namecallInstance")
            and type(rawget(v, "newindexInstance")) == "table"
            and v["newindexInstance"][1] == "kick" 
        then
            Hook_Adonis(v)
        end
    end
end

    while true do
        task.wait(1)
        if oreDropdown then
            local newOres = getAvailableOres()
            if #newOres > 0 then
                oreDropdown.Options = newOres
                if not table.find(oreDropdown.Options, oreDropdown.CurrentOption[1]) then
                    oreDropdown.CurrentOption = {newOres[1]}
                    oreDropdown:Set(oreDropdown.CurrentOption)
                end
            end
        end
    end
end)
