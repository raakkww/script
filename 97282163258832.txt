        wait(0.5)

            currentDots = currentDots + 1

            if currentDots >= maxDots then

                dotsDirection = -1

            end

        else

            currentDots = currentDots - 1

            if currentDots <= 0 then

                dotsDirection = 1

            end

        end

        

        local dots = ""

        for i = 1, currentDots do

            dots = dots .. "."

        end

        dotsLabel.Text = dots

    end

    

    loadingGui:Destroy()

end)

-- Функция NoClip

local function enableNoClip()

    local char = player.Character

    if not char then return end

    

    for _, part in pairs(char:GetDescendants()) do

        if part:IsA("BasePart") then

            part.CanCollide = false

        end

    end

end

-- Функции телепорта с NoClip

local function teleportToFlag()

    local char = player.Character

    if char and char:FindFirstChild("HumanoidRootPart") then

        local flag = workspace.map and workspace.map:FindFirstChild("endzone")

        if flag then

            enableNoClip()

            char:MoveTo(flag.Position)

            status.Text = "✅ FLAG"

            wait(1)

            status.Text = "⚫ READY"

        else

            status.Text = "❌ NO FLAG"

        end

    end

end

local function teleportToCoin(coin)

    local char = player.Character

    if char and char:FindFirstChild("HumanoidRootPart") and coin then

        local lastPos = char.HumanoidRootPart.Position

        enableNoClip()

        char:MoveTo(coin.Position)

        wait(0.3)

        if autoReturn then

            enableNoClip()

            char:MoveTo(lastPos)

        end

    end

end

local function scanForCoins()

    coins = {}

    

    for _, child in pairs(listFrame:GetChildren()) do

        if child:IsA("TextButton") then

            child:Destroy()

        end

    end

    

    for _, obj in pairs(workspace:GetDescendants()) do

        if obj:IsA("BasePart") and (obj.Name:lower():match("coin") or obj.Name:lower():match("money")) then

            table.insert(coins, obj)

        end

    end

    

    if #coins > 0 then

        listFrame.Visible = true

        tpAllBtn.Visible = true

        

        local yPos = 0

        for i, coin in ipairs(coins) do

            local coinBtn = Instance.new("TextButton")

            coinBtn.Text = "💰 Coin " .. i

            coinBtn.Size = UDim2.new(1, -10, 0, 22)

            coinBtn.Position = UDim2.new(0, 5, 0, yPos)

            coinBtn.BackgroundColor3 = Color3.fromRGB(60, 60, 70)

            coinBtn.TextColor3 = Color3.new(1, 1, 1)

            coinBtn.Font = Enum.Font.Gotham

            coinBtn.TextSize = 9

            coinBtn.TextXAlignment = Enum.TextXAlignment.Left

            coinBtn.Parent = listFrame

            

            local btnCorner = Instance.new("UICorner")

            btnCorner.CornerRadius = UDim.new(0, 3)

            btnCorner.Parent = coinBtn

            

            coinBtn.MouseButton1Click:Connect(function()

                teleportToCoin(coin)

                status.Text = "✅ DONE"

                wait(1)

                status.Text = "⚫ READY"

            end)

            

            yPos = yPos + 27

        end

        

        listFrame.CanvasSize = UDim2.new(0, 0, 0, yPos)

    else

        listFrame.Visible = false

        tpAllBtn.Visible = false

    end

end

tpAllBtn.MouseButton1Click:Connect(function()

    if #coins == 0 then return end

    status.Text = "🟡 TP ALL..."

    for i, coin in ipairs(coins) do

        teleportToCoin(coin)

        wait(0.2)

    end

    status.Text = "✅ ALL DONE"

    wait(1)

    status.Text = "⚫ READY"

end)

flagBtn.MouseButton1Click:Connect(teleportToFlag)

-- Автоматический скан каждые 3 секунды

spawn(function()

    while true do

        scanForCoins()
