--- 脨聸脨戮脨鲁脨赂脨潞脨掳 脨驴脨碌脩聙脨碌脩聜脨掳脩聛脨潞脨赂脨虏脨掳脨陆脨赂脩聫 (脩聜脨碌脨驴脨碌脩聙脩聦 脨路脨掳 mainFrame) ---
local dragging, dragInput, dragStart, startPos
mainFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = mainFrame.Position
    end
end)
UserInputService.InputChanged:Connect(function(input)
    if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
        local delta = input.Position - dragStart
        mainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)
UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then dragging = false end
end)

--- 脨聸脨戮脨鲁脨赂脨潞脨掳 脨虏脩聥脨卤脨戮脩聙脨掳 脩聛脨陆脨掳脩聙脩聫脨麓脨掳 ---
selectButton.MouseButton1Click:Connect(function()
    currentAmmoIndex = currentAmmoIndex + 1
    if currentAmmoIndex > #ammoTypes then currentAmmoIndex = 1 end
    ammoLabel.Text = "脨垄脨聵脨聼: " .. ammoTypes[currentAmmoIndex]
end)

--- 脨聸脨戮脨鲁脨赂脨潞脨掳 脨掳脩聜脨掳脨潞脨赂 ---
local isNuking = false
local lp = Players.LocalPlayer

button.MouseButton1Click:Connect(function()
    isNuking = not isNuking
    if isNuking then
        button.Text = "脨聬脨 脨垄脨聬: on"
        button.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
        
        task.spawn(function()
            while isNuking do
                local char = lp.Character
                local root = char and char:FindFirstChild("HumanoidRootPart")
                if root then
                    local p = root.Position
                    for i = 1, 30 do 
                        if not isNuking then break end
                        local targetPos = vector.create(
                            p.X + math.random(-150, 150),
                            p.Y - 5, 
                            p.Z + math.random(-150, 150)
                        )
                        -- 脨聵脩聛脨驴脨戮脨禄脩聦脨路脩聝脨碌脨录 脨虏脩聥脨卤脩聙脨掳脨陆脨陆脩聥脨鹿 脩聜脨赂脨驴 脩聛脨陆脨掳脩聙脩聫脨麓脨掳
                        remote:FireServer(ammoTypes[currentAmmoIndex], targetPos)
                    end
                end
                task.wait(0.05)
            end
        end)
    else
        button.Text = "脨聬脨 脨垄脨聬: off"
        button.BackgroundColor3 = Color3.fromRGB(150, 0, 0)
    end
end)
