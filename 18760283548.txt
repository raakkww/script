getgenv().StrengthLoop = false

-- Teleport function (Exact CFrame positions)
local player = game.Players.LocalPlayer
local function tpToPosition(x, y, z, areaName)
   if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
      player.Character.HumanoidRootPart.CFrame = CFrame.new(x, y, z) + Vector3.new(0, 5, 0)
      return true
   end
   return false
end

-- Tab 1: Strength
local MainTab = Window:CreateTab("√∞¬ü¬å¬Ä Strength", 4483362458)
local SectionLoops = MainTab:CreateSection("√∞¬ü¬î¬Ñ Auto Farm")

MainTab:CreateToggle({
   Name = "√∞¬ü¬í¬™ Auto Infinite Strength Loop",
   CurrentValue = false,
   Flag = "AutoStrength",
   Callback = function(Value)
      getgenv().StrengthLoop = Value
      if Value then
         StrengthConnection = game:GetService("RunService").Heartbeat:Connect(function()
            if getgenv().StrengthLoop then
               pcall(function()
                  game:GetService("ReplicatedStorage"):WaitForChild("StrengthIncrease"):FireServer()
               end)
            end
         end)
         Rayfield:Notify({
            Title = "Infinite Strength ACTIVE",
            Content = "Spamming StrengthIncrease... (Toggle OFF to stop)",
            Duration = 3
         })
      else
         if StrengthConnection then
            StrengthConnection:Disconnect()
            StrengthConnection = nil
         end
         Rayfield:Notify({
            Title = "Infinite Strength STOPPED",
            Content = "Loop disconnected.",
            Duration = 2
         })
      end
   end
})
