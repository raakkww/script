-- Game Services
local Players = game:GetService("Players")

local LocalPlayer = Players.LocalPlayer

local WeaponPackets = require(ReplicatedStorage.Common.Packets.WeaponPackets)

local Settings = SilentAimSettings

-- FOV Circle
local fovCircle = Drawing.new("Circle")

fovCircle.Thickness = 1
fovCircle.Transparency = 0.7

RunService.RenderStepped:Connect(function()
    fovCircle.Visible = Settings.Enabled and Settings.ShowFOV
    fovCircle.Position = Camera.ViewportSize / 2
    fovCircle.Radius = Settings.FOV
end)

-- Targeting
local function getClosestTarget()
    local screenCenter = Camera.ViewportSize / 2
    local closestPlayer, closestPart
    local closestDist = math.huge

    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character then
            local humanoid = player.Character:FindFirstChildOfClass("Humanoid")
            if humanoid and humanoid.Health > 0 then
                local part = player.Character:FindFirstChild(Settings.TargetPart)
                    or player.Character:FindFirstChild("Head")
                    or player.Character:FindFirstChild("HumanoidRootPart")

                if part then
                    local screenPos = Camera:WorldToViewportPoint(part.Position)
                    local dist = (Vector2.new(screenPos.X, screenPos.Y) - screenCenter).Magnitude

                    if dist <= Settings.FOV and dist < closestDist then
                        closestDist = dist
                        closestPlayer = player
                        closestPart = part
                    end
                end
            end
        end
    end

    return closestPlayer, closestPart
end

local function shouldHit()
    return math.random(1, 100) <= Settings.HitChance
end

-- Packet Intercept
local originalSend = WeaponPackets.useWeapon.send

WeaponPackets.useWeapon.send = function(data)
    if Settings.Enabled and shouldHit() then
        local target, part = getClosestTarget()
        if target and part then
            local newData = {
                hitResult = part.Parent,
                hitPart = part,
                position = part.Position,
                scoped = data.scoped,
                was360 = data.was360,
                quickscope = data.quickscope,
                weaponType = data.weaponType
            }
            return originalSend(newData)
        end
    end
    return originalSend(data)
end

-- Gun Tracker
local GunTracker = Drawing.new("Line")
GunTracker.Thickness = 2
GunTracker.Color = Color3.fromRGB(0, 255, 0)
GunTracker.Visible = false

RunService.RenderStepped:Connect(function()
    if not Toggles.GunTrackerEnabled or not Toggles.GunTrackerEnabled.Value or not Settings.Enabled then
        GunTracker.Visible = false
        return
    end

    local target, part = getClosestTarget()
    if not (target and part) then
        GunTracker.Visible = false
        return
    end

    local center = Camera.ViewportSize / 2
    local pos, onScreen = Camera:WorldToViewportPoint(part.Position)

    if onScreen then
        GunTracker.From = center
        GunTracker.To = Vector2.new(pos.X, pos.Y)
        GunTracker.Visible = true
    else
        GunTracker.Visible = false
    end
end)

-- Classic ESP (Fixed Dead Removal)
local ESP = {}

local function createESP(player)
    if ESP[player] then return end

    local box = Drawing.new("Square")
    box.Thickness = 1
    box.Filled = false
    box.Color = Color3.fromRGB(255, 0, 0)
    box.Visible = false

    local nameTag = Drawing.new("Text")
    nameTag.Size = 14
    nameTag.Center = true
    nameTag.Outline = true
    nameTag.Color = Color3.fromRGB(255, 255, 255)
    nameTag.Visible = false

    local distanceTag = Drawing.new("Text")
    distanceTag.Size = 13
    distanceTag.Center = true
    distanceTag.Outline = true
    distanceTag.Color = Color3.fromRGB(180, 180, 180)
    distanceTag.Visible = false

    ESP[player] = {
        Box = box,
        Name = nameTag,
        Distance = distanceTag
    }
end

Players.PlayerRemoving:Connect(function(player)
    if ESP[player] then
        for _, obj in pairs(ESP[player]) do
            obj:Remove()
        end
        ESP[player] = nil
    end
end)

RunService.RenderStepped:Connect(function()
    if not Toggles.ESPEnabled or not Toggles.ESPEnabled.Value then return end

    for _, player in ipairs(Players:GetPlayers()) do
        if player ~= LocalPlayer then
            local char = player.Character
            local hrp = char and char:FindFirstChild("HumanoidRootPart")
            local head = char and char:FindFirstChild("Head")
            local hum = char and char:FindFirstChildOfClass("Humanoid")

            if not (char and hrp and head and hum) or hum.Health <= 0 then
                if ESP[player] then
                    ESP[player].Box.Visible = false
                    ESP[player].Name.Visible = false
                    ESP[player].Distance.Visible = false
                end
                continue
            end

            createESP(player)

            local box = ESP[player].Box
            local nameTag = ESP[player].Name
            local distanceTag = ESP[player].Distance

            local hrpPos, onScreen = Camera:WorldToViewportPoint(hrp.Position)
            local headPos = Camera:WorldToViewportPoint(head.Position + Vector3.new(0, 0.5, 0))
            local feetPos = Camera:WorldToViewportPoint(hrp.Position - Vector3.new(0, 2.5, 0))

            if onScreen then
                local boxHeight = math.abs(headPos.Y - feetPos.Y)
                local boxWidth = boxHeight / 2

                box.Size = Vector2.new(boxWidth, boxHeight)
                box.Position = Vector2.new(hrpPos.X - boxWidth / 2, headPos.Y)
                box.Visible = true

                nameTag.Text = player.Name
                nameTag.Position = Vector2.new(hrpPos.X, headPos.Y - 16)
                nameTag.Visible = true

                local distance = math.floor((Camera.CFrame.Position - hrp.Position).Magnitude)
                distanceTag.Text = tostring(distance) .. " studs"
                distanceTag.Position = Vector2.new(hrpPos.X, headPos.Y + boxHeight + 2)
                distanceTag.Visible = true
            else
                box.Visible = false
                nameTag.Visible = false
                distanceTag.Visible = false
            end
        end
    end
end)
