local function al(ch, ht)
    local head = ch:FindFirstChild("Head")
    if not head or not ht then
        return
    end
    local han = ht:FindFirstChild("Handle") or ht:FindFirstChildWhichIsA("BasePart")
    if not han then
        return
    end
    han.CanCollide = false
    local w = Instance.new("Weld", han)
    w.Name = "AccessoryWeld"
    w.Part0 = head
    w.Part1 = han
    local ha = han:FindFirstChildWhichIsA("Attachment")
    local hea = ha and head:FindFirstChild(ha.Name)
    w.C0 = (ha and hea) and hea.CFrame or CFrame.new(0, 0.6, 0)
    w.C1 = ha and ha.CFrame or CFrame.new(0, 0, 0)
    w.Parent = han
    ht.Parent = ch
end

local function giveFL()
    if lp.Backpack:FindFirstChild("Facility Flashlight") or lp.Character:FindFirstChild("Facility Flashlight") then
        return
    end
    local tool = Instance.new("Tool")
    tool.Name = "Facility Flashlight"
    tool.RequiresHandle = true
    tool.CanBeDropped = false
    local handle = Instance.new("Part")
    handle.Name = "Handle"
    handle.Size = Vector3.new(0.4, 0.4, 1.5)
    handle.Color = Color3.fromRGB(15, 15, 15)
    local light = Instance.new("SpotLight", handle)
    light.Brightness = 2.5
    light.Range = 75
    light.Enabled = false
    tool.Equipped:Connect(
        function()
            light.Enabled = true
            ws:Send(H:JSONEncode({keyword = kw, sender = lp.Name, command = "FLASHLIGHT_ON"}))
        end
    )
    tool.Unequipped:Connect(
        function()
            light.Enabled = false
            ws:Send(H:JSONEncode({keyword = kw, sender = lp.Name, command = "FLASHLIGHT_OFF"}))
        end
    )
    handle.Parent = tool
    tool.Parent = lp.Backpack
    return tool
end

local function addToLog(sName, msg)
    local COLORS = {
        Color3.new(1, 0.16, 0.26),
        Color3.new(0, 0.63, 1),
        Color3.new(0, 0.72, 0.34),
        Color3.fromRGB(180, 0, 255),
        Color3.fromRGB(255, 145, 0)
    }
    local function gv(n)
        local v = 0
        for i = 1, #n do
            v = v + string.byte(n:sub(i, i))
        end
        return v
    end
    local hex = COLORS[(gv(sName) % #COLORS) + 1]:ToHex()
    local row = Instance.new("Frame", ChatDisplay)
    row.Size = UDim2.new(1, 0, 0, 0)
    row.AutomaticSize = "Y"
    row.BackgroundTransparency = 1
    local l = Instance.new("TextLabel", row)
    l.Size = UDim2.new(1, -45, 0, 0)
    l.AutomaticSize = "Y"
    l.BackgroundTransparency = 1
    l.TextColor3 = Color3.new(0, 0, 0)
    l.RichText = true
    l.TextWrapped = true
    l.TextXAlignment = "Left"
    l.TextSize = 14
    l.RichText = true
    l.Text = "<b>[<font color='#" .. hex .. "'>" .. sName .. "</font>]</b>: " .. msg
    local cp = Instance.new("TextButton", row)
    cp.Text = "Copy"
    cp.Size = UDim2.new(0, 35, 0, 18)
    cp.Position = UDim2.new(1, -40, 0, 0)
    cp.Activated:Connect(
        function()
            (set_clipboard or print)(msg)
        end
    )
    ChatDisplay.CanvasPosition = Vector2.new(0, 9999)
end

ws.OnMessage:Connect(
    function(msg)
        local s, d =
            pcall(
            function()
                return H:JSONDecode(msg)
            end
        )
        if not s then
            local n, content = msg:match("^(.-):%s*(.*)$")
            if n then
                addToLog(n, content)
                local p = P:FindFirstChild(n)
                if p and p.Character then
                    C:Chat(p.Character.Head, content, "White")
                end
            end
            return
        end
        if d.keyword == kw and d.sender ~= lp.Name then
            local tp = P:FindFirstChild(d.sender)

            -- [NEW SYNC COMMAND HANDLER] (Ooooh what you doing you checking my code haha, i see you! :D also this is just Give all Tools Sync to all players who have the script.)
            if d.command == "SYNC_GIVE_TOOLS" then
                grabAllItems()
                return
            end

            if not tp or not tp.Character then
                return
            end
            if d.command == "FLASHLIGHT_ON" then
                local hand = tp.Character:FindFirstChild("RightHand") or tp.Character:FindFirstChild("Right Arm")
                if hand then
                    if tp.Character:FindFirstChild("SyncFL") then
                        tp.Character.SyncFL:Destroy()
                    end
                    local f = Instance.new("Part", tp.Character)
                    f.Name = "SyncFL"
                    f.Size = Vector3.new(0.4, 0.4, 1.5)
                    f.Color = Color3.new(0, 0, 0)
                    f.CanCollide = false
                    local l = Instance.new("SpotLight", f)
                    l.Brightness = 2.5
                    l.Range = 75
                    local w = Instance.new("Weld", f)
                    w.Part0 = hand
                    w.Part1 = f
                    w.C0 = CFrame.new(0, -1, 0) * CFrame.Angles(math.rad(-90), 0, 0)
                end
            elseif d.command == "FLASHLIGHT_OFF" then
                if tp.Character:FindFirstChild("SyncFL") then
                    tp.Character.SyncFL:Destroy()
                end
            elseif d.command == "CLEAR" then
                for _, v in pairs(tp.Character:GetDescendants()) do
                    if v.Name == "AccessoryWeld" then
                        v.Parent.Parent:Destroy()
                    elseif v.Name == "SyncFL" then
                        v:Destroy()
                    end
                end
            elseif d.hatName then
                local h = dh:FindFirstChild(d.hatName)
                if h then
                    al(tp.Character, h:Clone())
                end
            end
        end
    end
)

ChatInput.FocusLost:Connect(
    function(e)
        if e and ChatInput.Text ~= "" then
            local currentMessage = ChatInput.Text
            if currentMessage == lastMsg then
                S:SetCore(
                    "SendNotification",
                    {Title = "⚠️ DON'T SPAM", Text = "Duplicate message detected!", Duration = 3}
                )
                task.wait(dbTime)
                ChatInput.Text = ""
                return
            end
            ws:Send(lp.Name .. ": " .. currentMessage)
            addToLog(lp.Name, currentMessage)
            if lp.Character then
                C:Chat(lp.Character.Head, currentMessage, "White")
            end
            lastMsg = currentMessage
            ChatInput.Text = ""
        end
    end
)

local function Terminate(keepHats)
    if not keepHats then
        ws:Send(H:JSONEncode({keyword = kw, sender = lp.Name, command = "CLEAR"}))
        if lp.Character then
            for _, v in pairs(lp.Character:GetDescendants()) do
                if v.Name == "AccessoryWeld" then
                    v.Parent.Parent:Destroy()
                end
            end
        end
    end
    if ncloop then
        ncloop:Disconnect()
    end
    if lp.Character then
        for _, v in pairs(lp.Character:GetDescendants()) do
            if v:IsA("BasePart") then
                v.CanCollide = true
            end
        end
    end
    if fl_tool then
        fl_tool:Destroy()
    end
    _G.BleachLoaded = nil
    g:Destroy()
end
