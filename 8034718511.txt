local Players = game:GetService("Players")
local Player = Players.LocalPlayer

local Character = Player.Character or Player.CharacterAdded:Wait()
local Root = Character:WaitForChild("HumanoidRootPart")

Player.CharacterAdded:Connect(function(char)
    Character = char
    Root = char:WaitForChild("HumanoidRootPart")
end)

local function GetOwnedTycoon()
    for _, tycoon in pairs(workspace.Treehouses:GetChildren()) do
        if tycoon:IsA("Model") then
            
            local owner = tycoon:FindFirstChild("Owner")
            
            if owner then
                if owner:IsA("ObjectValue") and owner.Value == Player then
                    return tycoon
                end
                if owner:IsA("StringValue") and owner.Value == Player.Name then
                    return tycoon
                end
                if owner:IsA("IntValue") and owner.Value == Player.UserId then
                    return tycoon
                end
            end
        end
    end
end

local function GetButtonPart()
    local tycoon = GetOwnedTycoon()
    if not tycoon then return nil end

    local success, part = pcall(function()
        return tycoon.Essentials.Giver.Button.Head
    end)

    if success then return part end
    return nil
end

local function CreateCopyLabel(Text, Value, YPos)
    local Label = Instance.new("TextButton", Frame)
    Label.Size = UDim2.new(1,-20,0,30)
    Label.Position = UDim2.new(0,10,0,YPos)
    Label.BackgroundTransparency = 1
    Label.Font = Enum.Font.Code
    Label.TextScaled = true
    Label.RichText = true

    local BaseText = '<font color="rgb(255,255,255)">'..Text..'</font> <font color="rgb(0,170,255)">'..Value..'</font>'
    Label.Text = BaseText

    Label.MouseEnter:Connect(function()
        Label.Text = BaseText .. ' <font color="rgb(150,150,150)">(Click to Copy)</font>'
    end)

    Label.MouseLeave:Connect(function()
        Label.Text = BaseText
    end)

    Label.MouseButton1Click:Connect(function()
        if setclipboard then
            setclipboard(Value)
        end
    end)
end

local Toggle = Instance.new("TextButton", Frame)
Toggle.Size = UDim2.new(1,-20,0,40)
Toggle.Position = UDim2.new(0,10,1,-50)
Toggle.Text = "Auto Collect: OFF"
Toggle.BackgroundColor3 = Color3.fromRGB(35,35,35)
Toggle.TextColor3 = Color3.new(1,1,1)
Toggle.Font = Enum.Font.GothamBold
Toggle.TextScaled = true

Instance.new("UICorner", Toggle).CornerRadius = UDim.new(0,8)

Toggle.MouseButton1Click:Connect(function()
    Running = not Running

    if Running then
        Toggle.Text = "Auto Collect: ON"
        Stroke.Color = Color3.fromRGB(0,255,120)

        ButtonPart = GetButtonPart()

        if ButtonPart then
            ButtonPart.CanCollide = false
            OriginalCFrame = ButtonPart.CFrame
        end

    else
        Toggle.Text = "Auto Collect: OFF"
        Stroke.Color = Color3.fromRGB(0,170,255)

        if ButtonPart and OriginalCFrame then
            ButtonPart.CFrame = OriginalCFrame
        end
    end
end)

task.spawn(function()
    while true do
        task.wait()

        if Running then
            if not ButtonPart or not ButtonPart.Parent then
                ButtonPart = GetButtonPart()
                if ButtonPart then
                    OriginalCFrame = ButtonPart.CFrame
                end
            end

            if ButtonPart and Root then
                ButtonPart.CFrame = Root.CFrame
            end
        end
    end
end)
