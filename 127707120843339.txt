local function sendChatMessage(message)
    if TextChatService.ChatVersion == Enum.ChatVersion.TextChatService then
        local textChannel = TextChatService.TextChannels:FindFirstChild("RBXGeneral")
        if textChannel then
            textChannel:SendAsync(tostring(message))
        end
    else
        game.ReplicatedStorage.DefaultChatSystemChatEvents.SayMessageRequest:FireServer(tostring(message), "All")
    end
end

AutoSolverButton.MouseButton1Click:Connect(function()
    autoSolverEnabled = not autoSolverEnabled
    
    if autoSolverEnabled then
        AutoSolverButton.Text = "Auto Solver [ON]"
        AutoSolverButton.BackgroundColor3 = Color3.fromRGB(50, 200, 50)
        
        if autoSolverConnection then
            autoSolverConnection:Disconnect()
        end
        
        autoSolverConnection = pcTextBoxContainer:GetPropertyChangedSignal("Visible"):Connect(function()
            if not pcTextBoxContainer.Visible or not autoSolverEnabled then return end
            
            local question = questionText.Text
            question = question:gsub("x", "*"):gsub("=", "")
            
            local func = loadstring("return " .. question)
            if func then
                local success, answer = pcall(func)
                pcTextBoxContainer.TextBox.Text = ""
                if success then
                    task.wait(0.5)
                    for i = 1, #tostring(answer) do
                        task.wait(math.random())
                        pcTextBoxContainer.TextBox.Text ..= tostring(answer):sub(i, i)
                    end
                    task.wait(0.5)
                    VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.Return, false, game)
                    task.wait(0.05)
                    VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.Return, false, game)
                end
            end
        end)
    else
        AutoSolverButton.Text = "Auto Solver [OFF]"
        AutoSolverButton.BackgroundColor3 = Color3.fromRGB(200, 50, 50)
        
        if autoSolverConnection then
            autoSolverConnection:Disconnect()
            autoSolverConnection = nil
        end
    end
end)

local function applyNoCooldown(tool)
    if tool.Name ~= "Tomato" then return end
    
    if tool:FindFirstChild("Cooldown") then
        tool.Cooldown:Destroy()
    end
    
    local cooldownValue = tool:FindFirstChild("CooldownValue")
    if cooldownValue and cooldownValue:IsA("NumberValue") then
        cooldownValue.Value = 0
    end
    
    for _, descendant in pairs(tool:GetDescendants()) do
        if descendant:IsA("NumberValue") then
            if descendant.Name:lower():find("cooldown") or descendant.Name:lower():find("delay") or descendant.Name:lower():find("wait") then
                descendant.Value = 0
                
                table.insert(noCooldownConnections, descendant:GetPropertyChangedSignal("Value"):Connect(function()
                    if noCooldownEnabled and descendant.Value > 0 then
                        descendant.Value = 0
                    end
                end))
            end
        elseif descendant:IsA("IntValue") then
            if descendant.Name:lower():find("cooldown") or descendant.Name:lower():find("delay") or descendant.Name:lower():find("wait") then
                descendant.Value = 0
                
                table.insert(noCooldownConnections, descendant:GetPropertyChangedSignal("Value"):Connect(function()
                    if noCooldownEnabled and descendant.Value > 0 then
                        descendant.Value = 0
                    end
                end))
            end
        end
    end
    
    table.insert(noCooldownConnections, tool.ChildAdded:Connect(function(child)
        if noCooldownEnabled and child.Name == "Cooldown" then
            task.wait()
            child:Destroy()
        end
    end))
    
    table.insert(noCooldownConnections, tool.Activated:Connect(function()
        if noCooldownEnabled then
            task.spawn(function()
                task.wait(0.05)
                if tool:FindFirstChild("Cooldown") then
                    tool.Cooldown:Destroy()
                end
                for _, descendant in pairs(tool:GetDescendants()) do
                    if descendant:IsA("NumberValue") or descendant:IsA("IntValue") then
                        if descendant.Name:lower():find("cooldown") or descendant.Name:lower():find("delay") or descendant.Name:lower():find("wait") then
                            descendant.Value = 0
                        end
                    end
                end
            end)
        end
    end))
end

TomatoButton.MouseButton1Click:Connect(function()
    noCooldownEnabled = not noCooldownEnabled
    
    if noCooldownEnabled then
        TomatoButton.Text = "No Cooldown Tomato [ON]"
        TomatoButton.BackgroundColor3 = Color3.fromRGB(50, 200, 50)
        
        local character = player.Character or player.CharacterAdded:Wait()
        local backpack = player.Backpack
        
        for _, tool in pairs(backpack:GetChildren()) do
            if tool:IsA("Tool") then
                applyNoCooldown(tool)
            end
        end
        
        for _, tool in pairs(character:GetChildren()) do
            if tool:IsA("Tool") then
                applyNoCooldown(tool)
            end
        end
        
        table.insert(noCooldownConnections, backpack.ChildAdded:Connect(function(tool)
            if noCooldownEnabled and tool:IsA("Tool") then
                task.wait(0.1)
                applyNoCooldown(tool)
            end
        end))
        
        table.insert(noCooldownConnections, character.ChildAdded:Connect(function(tool)
            if noCooldownEnabled and tool:IsA("Tool") then
                task.wait(0.1)
                applyNoCooldown(tool)
            end
        end))
        
        table.insert(noCooldownConnections, player.CharacterAdded:Connect(function(newChar)
            if noCooldownEnabled then
                task.wait(1)
                for _, tool in pairs(newChar:GetChildren()) do
                    if tool:IsA("Tool") then
                        applyNoCooldown(tool)
                    end
                end
                
                table.insert(noCooldownConnections, newChar.ChildAdded:Connect(function(tool)
                    if noCooldownEnabled and tool:IsA("Tool") then
                        task.wait(0.1)
                        applyNoCooldown(tool)
                    end
                end))
            end
        end))
        
        task.spawn(function()
            while noCooldownEnabled do
                task.wait(0.5)
                local char = player.Character
                if char then
                    for _, tool in pairs(char:GetChildren()) do
                        if tool:IsA("Tool") and tool.Name == "Tomato" then
                            if tool:FindFirstChild("Cooldown") then
                                tool.Cooldown:Destroy()
                            end
                        end
                    end
                end
                
                for _, tool in pairs(backpack:GetChildren()) do
                    if tool:IsA("Tool") and tool.Name == "Tomato" then
                        if tool:FindFirstChild("Cooldown") then
                            tool.Cooldown:Destroy()
                        end
                    end
                end
            end
        end)
        
    else
        TomatoButton.Text = "No Cooldown Tomato [OFF] (PATCHED)"
        TomatoButton.BackgroundColor3 = Color3.fromRGB(200, 50, 50)
        
        for _, connection in pairs(noCooldownConnections) do
            if connection and connection.Connected then
                connection:Disconnect()
            end
        end
        noCooldownConnections = {}
    end
end)
