local a = loadstring(game:HttpGet("https://sirius.menu/rayfield"))()
local b = game:GetService("Players").LocalPlayer
local c = b:GetMouse()
local d = workspace.CurrentCamera
local e = game:GetService("UserInputService")
local f, g, h = nil, nil, nil
local i = false
local j = false
local function k(l)
    if not l then
        return
    end
    l:SetStateEnabled(Enum.HumanoidStateType.FallingDown, false)
    l:SetStateEnabled(Enum.HumanoidStateType.Ragdoll, false)
    l:ChangeState(Enum.HumanoidStateType.GettingUp)
end
c.Button1Down:Connect(
    function()
        if j and e:IsKeyDown(Enum.KeyCode.LeftShift) then
            local m = b.Character and b.Character:FindFirstChild("HumanoidRootPart")
            if m then
                m.CFrame = CFrame.new(c.Hit.p + Vector3.new(0, 3, 0))
            end
        end
    end
)
local function n()
    local o = b.Character
    local p = o and o:FindFirstChild("Humanoid")
    local m = o and o:FindFirstChild("HumanoidRootPart")
    if not m or not p then
        return
    end
    k(p)
    local q = nil
    for r, s in pairs(workspace:GetDescendants()) do
        if s:IsA("ProximityPrompt") then
            local t = (s.Parent.Position - m.Position).Magnitude
            if t < 25 then
                q = s
                break
            end
        end
    end
    if q then
        m.Anchored = true
        d.CameraType = Enum.CameraType.Scriptable
        d.CFrame = CFrame.new(d.CFrame.Position, q.Parent.Position)
        task.wait(0.1)
        q:InputHoldBegin()
        task.wait(2.2)
        q:InputHoldEnd()
        d.CameraType = Enum.CameraType.Custom
        m.Anchored = false
        return true
    end
end
local function u(v)
    local m = b.Character and b.Character:FindFirstChild("HumanoidRootPart")
    if not m or not v then
        return
    end
    while i and (m.Position - v).Magnitude > 3 do
        local w = m.Position
        local x = (v - w).Unit
        m.CFrame = CFrame.new(w + x * 7, Vector3.new(v.X, w.Y, v.Z))
        m.Velocity = Vector3.new(0, 0, 0)
        task.wait(0.1)
    end
end
