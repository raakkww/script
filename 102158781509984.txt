local ScreenGui = Instance.new("ScreenGui", LocalPlayer:WaitForChild("PlayerGui"))
local MainFrame = Instance.new("Frame", ScreenGui)
MainFrame.Size = UDim2.new(0, 200, 0, 100)
MainFrame.Position = UDim2.new(0.5, -100, 0.5, -50)
MainFrame.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
MainFrame.Active = true
MainFrame.Draggable = true

local ToggleButton = Instance.new("TextButton", MainFrame)
ToggleButton.Size = UDim2.new(0, 160, 0, 40)
ToggleButton.Position = UDim2.new(0.5, -80, 0.5, -20)
ToggleButton.Text = "TP Collect: OFF"
ToggleButton.BackgroundColor3 = Color3.fromRGB(200, 50, 50)
ToggleButton.TextColor3 = Color3.new(1, 1, 1)

local active = false
local cratesFolder = workspace:WaitForChild("Map"):WaitForChild("Crates")

ToggleButton.MouseButton1Click:Connect(function()
    active = not active
    ToggleButton.Text = active and "TP Collect: ON" or "TP Collect: OFF"
    ToggleButton.BackgroundColor3 = active and Color3.fromRGB(50, 200, 50) or Color3.fromRGB(200, 50, 50)
end)

task.spawn(function()
    while true do
        if active and LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
            local root = LocalPlayer.Character.HumanoidRootPart
            local models = cratesFolder:GetChildren()
            
            for _, model in ipairs(models) do
                if not active then break end
                
                local target = model:FindFirstChild("CollectPart")
                if target and target:IsA("BasePart") then

                    local oldPos = root.CFrame
                    
                    root.CFrame = target.CFrame
                    task.wait(0.05)
                    firetouchinterest(root, target, 0)
                    firetouchinterest(root, target, 1)
                    
                    task.wait(0.05)
                    root.CFrame = oldPos
                    
                    task.wait(0.25)
                end
            end
        end
        task.wait(0.1)
    end
end)
