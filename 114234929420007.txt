local function GetConfigList()
    local configs = {}
    local success, files = pcall(function()
        return listfiles()
    end)
    
    if success then
        for _, file in pairs(files) do
            if file:match("BloXtrike_(.+)%.json") then
                local name = file:match("BloXtrike_(.+)%.json")
                table.insert(configs, name)
            end
        end
    end
    
    return configs
end

-- GET BLOXSTRIKE PLAYERS
---------------------------------------------------------------------
local function GetBloXtrikePlayers()
    local players = {}
    
    -- BloXtrike stores players in workspace differently
    local playersFolder = Workspace:FindFirstChild("Players") or Workspace:FindFirstChild("Characters")
    
    if playersFolder then
        for _, playerModel in pairs(playersFolder:GetChildren()) do
            if playerModel:IsA("Model") and playerModel ~= LocalPlayer.Character then
                local head = playerModel:FindFirstChild("Head")
                local humanoid = playerModel:FindFirstChildOfClass("Humanoid")
                
                if head and humanoid and humanoid.Health > 0 then
                    -- Get actual player object
                    local playerName = playerModel.Name
                    local player = Players:FindFirstChild(playerName)
                    
                    if player then
                        table.insert(players, {
                            Player = player,
                            Character = playerModel,
                            Head = head,
                            Humanoid = humanoid
                        })
                    end
                end
            end
        end
    end
    
    -- Also check standard player characters
    for _, player in pairs(Players:GetPlayers()) do
        if player ~= LocalPlayer and player.Character then
            local char = player.Character
            local head = char:FindFirstChild("Head")
            local humanoid = char:FindFirstChildOfClass("Humanoid")
            
            if head and humanoid and humanoid.Health > 0 then
                local alreadyAdded = false
                for _, p in pairs(players) do
                    if p.Player == player then
                        alreadyAdded = true
                        break
                    end
                end
                
                if not alreadyAdded then
                    table.insert(players, {
                        Player = player,
                        Character = char,
                        Head = head,
                        Humanoid = humanoid
                    })
                end
            end
        end
    end
    
    return players
end

local function IsOnSameTeam(targetModel)
    if not _G.TeamCheck then return false end
    
    -- Try to find player by model name
    local targetPlayer = Players:FindFirstChild(targetModel.Name)
    if targetPlayer and targetPlayer.Team and LocalPlayer.Team then
        return targetPlayer.Team == LocalPlayer.Team
    end
    
    -- Fallback: check team color on character (BloXtrike method)
    local targetTeamColor = targetModel:FindFirstChild("TeamColor")
    local myChar = LocalPlayer.Character
    local myTeamColor = myChar and myChar:FindFirstChild("TeamColor")
    
    if targetTeamColor and myTeamColor then
        return targetTeamColor.Value == myTeamColor.Value
    end
    
    return false
end

local function GetClosestPlayerInFOV()
    local closestPlayer = nil
    local shortestDistance = _G.AimbotFOV
    
    -- Try BloXtrike's camera children method (most common)
    for _, model in pairs(Camera:GetChildren()) do
        if model:IsA("Model") and model.Name ~= LocalPlayer.Name then
            -- Skip teammates
            if IsOnSameTeam(model) then
                continue
            end
            
            local head = model:FindFirstChild("Head") or model:FindFirstChild("HeadHB")
            if head and head:IsA("BasePart") then
                local screenPos, onScreen = Camera:WorldToViewportPoint(head.Position)
                if onScreen then
                    local mousePos = UserInputService:GetMouseLocation()
                    local distance = (Vector2.new(screenPos.X, screenPos.Y) - mousePos).Magnitude
                    
                    if distance < shortestDistance then
                        shortestDistance = distance
                        closestPlayer = head
                    end
                end
            end
        end
    end
    
    -- Fallback: check workspace players
    if not closestPlayer then
        local players = GetBloXtrikePlayers()
        for _, playerData in pairs(players) do
            local player = playerData.Player
            local head = playerData.Head
            
            if head then
                if _G.TeamCheck and player.Team == LocalPlayer.Team then
                    continue
                end
                
                local screenPos, onScreen = Camera:WorldToViewportPoint(head.Position)
                if onScreen then
                    local mousePos = UserInputService:GetMouseLocation()
                    local distance = (Vector2.new(screenPos.X, screenPos.Y) - mousePos).Magnitude
                    
                    if distance < shortestDistance then
                        shortestDistance = distance
                        closestPlayer = head
                    end
                end
            end
        end
    end
    
    return closestPlayer
end

-- CHAMS

local function ApplyChams(character)
    if ChamObjects[character] then return end
    
    -- Check if teammate
    if _G.TeamCheck then
        local player = Players:GetPlayerFromCharacter(character)
        if player and player.Team == LocalPlayer.Team then
            return -- Don't apply chams to teammates
        end
        
        -- Also check via model name
        local targetPlayer = Players:FindFirstChild(character.Name)
        if targetPlayer and targetPlayer.Team and LocalPlayer.Team then
            if targetPlayer.Team == LocalPlayer.Team then
                return
            end
        end
    end
    
    ChamObjects[character] = {}
    
    for _, part in pairs(character:GetDescendants()) do
        if part:IsA("BasePart") and part.Name ~= "HumanoidRootPart" then
            local highlight = Instance.new("Highlight")
            highlight.FillColor = Color3.fromRGB(255, 0, 0)
            highlight.OutlineColor = Color3.fromRGB(255, 255, 255)
            highlight.FillTransparency = 0.5
            highlight.OutlineTransparency = 0
            highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
            highlight.Parent = part
            
            table.insert(ChamObjects[character], highlight)
        end
    end
end

local function RemoveChams(character)
    if ChamObjects[character] then
        for _, highlight in pairs(ChamObjects[character]) do
            highlight:Destroy()
        end
        ChamObjects[character] = nil
    end
end

task.spawn(function()
    RunService.RenderStepped:Connect(function()
        if _G.ChamsEnabled then
            local players = GetBloXtrikePlayers()
            for _, playerData in pairs(players) do
                local player = playerData.Player
                local character = playerData.Character
                
                -- Skip teammates if team check enabled
                if _G.TeamCheck and player.Team == LocalPlayer.Team then
                    RemoveChams(character)
                else
                    ApplyChams(character)
                end
            end
        else
            for character, _ in pairs(ChamObjects) do
                RemoveChams(character)
            end
        end
    end)
end)

local function CreateESP(playerData)
    local character = playerData.Character
    if ESPObjects[character] then return end
    
    local esp = {
        Box = Drawing.new("Square"),
        BoxOutline = Drawing.new("Square"),
        Name = Drawing.new("Text"),
        Distance = Drawing.new("Text"),
        Health = Drawing.new("Text"),
        Tracer = Drawing.new("Line"),
        SkeletonLines = {}
    }
    
    -- Box
    esp.Box.Thickness = 1
    esp.Box.Filled = false
    esp.Box.Color = Color3.fromRGB(255, 255, 255)
    esp.Box.Transparency = 1
    esp.Box.Visible = false
    
    esp.BoxOutline.Thickness = 3
    esp.BoxOutline.Filled = false
    esp.BoxOutline.Color = Color3.fromRGB(0, 0, 0)
    esp.BoxOutline.Transparency = 1
    esp.BoxOutline.Visible = false
    
    -- Text
    esp.Name.Center = true
    esp.Name.Outline = true
    esp.Name.Color = Color3.fromRGB(255, 255, 255)
    esp.Name.Size = 13
    esp.Name.Visible = false
    
    esp.Distance.Center = true
    esp.Distance.Outline = true
    esp.Distance.Color = Color3.fromRGB(255, 255, 255)
    esp.Distance.Size = 13
    esp.Distance.Visible = false
    
    esp.Health.Center = false
    esp.Health.Outline = true
    esp.Health.Color = Color3.fromRGB(0, 255, 0)
    esp.Health.Size = 13
    esp.Health.Visible = false
    
    -- Tracer
    esp.Tracer.Thickness = 1
    esp.Tracer.Color = Color3.fromRGB(255, 255, 255)
    esp.Tracer.Transparency = 1
    esp.Tracer.Visible = false
    
    -- Skeleton lines
    local skeletonPairs = {
        {"Head", "UpperTorso"},
        {"UpperTorso", "LowerTorso"},
        {"UpperTorso", "LeftUpperArm"},
        {"LeftUpperArm", "LeftLowerArm"},
        {"LeftLowerArm", "LeftHand"},
        {"UpperTorso", "RightUpperArm"},
        {"RightUpperArm", "RightLowerArm"},
        {"RightLowerArm", "RightHand"},
        {"LowerTorso", "LeftUpperLeg"},
        {"LeftUpperLeg", "LeftLowerLeg"},
        {"LeftLowerLeg", "LeftFoot"},
        {"LowerTorso", "RightUpperLeg"},
        {"RightUpperLeg", "RightLowerLeg"},
        {"RightLowerLeg", "RightFoot"}
    }
    
    for i, pair in ipairs(skeletonPairs) do
        local line = Drawing.new("Line")
        line.Thickness = 1
        line.Color = Color3.fromRGB(255, 255, 255)
        line.Transparency = 1
        line.Visible = false
        esp.SkeletonLines[i] = {line = line, from = pair[1], to = pair[2]}
    end
    
    ESPObjects[character] = esp
end

local function RemoveESP(character)
    if ESPObjects[character] then
        ESPObjects[character].Box:Remove()
        ESPObjects[character].BoxOutline:Remove()
        ESPObjects[character].Name:Remove()
        ESPObjects[character].Distance:Remove()
        ESPObjects[character].Health:Remove()
        ESPObjects[character].Tracer:Remove()
        
        for _, skelData in pairs(ESPObjects[character].SkeletonLines) do
            skelData.line:Remove()
        end
        
        ESPObjects[character] = nil
    end
end

local function UpdateESP()
    local players = GetBloXtrikePlayers()
    local activeCharacters = {}
    
    for _, playerData in pairs(players) do
        local character = playerData.Character
        local player = playerData.Player
        local humanoid = playerData.Humanoid
        
        activeCharacters[character] = true
        
        if not ESPObjects[character] then
            CreateESP(playerData)
        end
        
        local esp = ESPObjects[character]
        local hrp = character:FindFirstChild("HumanoidRootPart")
        local head = character:FindFirstChild("Head")
        
        if hrp and head then
            local rootPos = hrp.Position
            local vector, onScreen = Camera:WorldToViewportPoint(rootPos)
            
            if onScreen then
                -- Simplified box calculation for performance
                local headPos = Camera:WorldToViewportPoint(head.Position + Vector3.new(0, 0.5, 0))
                local footPos = Camera:WorldToViewportPoint(hrp.Position - Vector3.new(0, 3, 0))
                
                local height = math.abs(headPos.Y - footPos.Y)
                local width = height / 2
                
                local minX = vector.X - width / 2
                local minY = headPos.Y
                local maxX = vector.X + width / 2
                local maxY = footPos.Y
                
                local boxWidth = maxX - minX
                local boxHeight = maxY - minY
                
                -- Team color
                local color = Color3.fromRGB(255, 255, 255)
                if player.Team then
                    color = player.Team.TeamColor.Color
                end
                
                -- Skip rendering if teammate and team check enabled
                local isTeammate = _G.TeamCheck and player.Team == LocalPlayer.Team
                
                if isTeammate then
                    -- Hide all ESP for teammates
                    esp.Box.Visible = false
                    esp.BoxOutline.Visible = false
                    esp.Name.Visible = false
                    esp.Distance.Visible = false
                    esp.Health.Visible = false
                    esp.Tracer.Visible = false
                    for _, skelData in pairs(esp.SkeletonLines) do
                        skelData.line.Visible = false
                    end
                else
                    -- Show ESP for enemies
                    -- Box ESP
                    if _G.BoxESP then
                    esp.Box.Size = Vector2.new(boxWidth, boxHeight)
                    esp.Box.Position = Vector2.new(minX, minY)
                    esp.Box.Color = color
                    esp.Box.Visible = true
                    
                    esp.BoxOutline.Size = Vector2.new(boxWidth, boxHeight)
                    esp.BoxOutline.Position = Vector2.new(minX, minY)
                    esp.BoxOutline.Visible = true
                else
                    esp.Box.Visible = false
                    esp.BoxOutline.Visible = false
                end
                
                -- Name ESP
                if _G.NameESP then
                    esp.Name.Text = player.Name
                    esp.Name.Position = Vector2.new(minX + boxWidth / 2, minY - 16)
                    esp.Name.Color = color
                    esp.Name.Visible = true
                else
                    esp.Name.Visible = false
                end
                
                -- Distance ESP
                if _G.DistanceESP and LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart") then
                    local distance = (LocalPlayer.Character.HumanoidRootPart.Position - rootPos).Magnitude
                    esp.Distance.Text = math.floor(distance) .. "m"
                    esp.Distance.Position = Vector2.new(minX + boxWidth / 2, maxY + 2)
                    esp.Distance.Visible = true
                else
                    esp.Distance.Visible = false
                end
                
                -- Health ESP
                if _G.HealthESP and humanoid then
                    local health = math.floor(humanoid.Health)
                    esp.Health.Text = tostring(health) .. " HP"
                    esp.Health.Position = Vector2.new(minX - 40, minY)
                    
                    local healthPercent = health / humanoid.MaxHealth
                    if healthPercent > 0.75 then
                        esp.Health.Color = Color3.fromRGB(0, 255, 0)
                    elseif healthPercent > 0.5 then
                        esp.Health.Color = Color3.fromRGB(255, 255, 0)
                    elseif healthPercent > 0.25 then
                        esp.Health.Color = Color3.fromRGB(255, 165, 0)
                    else
                        esp.Health.Color = Color3.fromRGB(255, 0, 0)
                    end
                    esp.Health.Visible = true
                else
                    esp.Health.Visible = false
                end
                
                -- Tracer ESP
                if _G.TracerESP then
                    esp.Tracer.From = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y)
                    esp.Tracer.To = Vector2.new(minX + boxWidth / 2, maxY)
                    esp.Tracer.Color = color
                    esp.Tracer.Visible = true
                else
                    esp.Tracer.Visible = false
                end
                
                -- Skeleton ESP
                if _G.SkeletonESP then
                    for _, skelData in pairs(esp.SkeletonLines) do
                        local fromPart = character:FindFirstChild(skelData.from)
                        local toPart = character:FindFirstChild(skelData.to)
                        
                        if fromPart and toPart then
                            local fromPos, fromOnScreen = Camera:WorldToViewportPoint(fromPart.Position)
                            local toPos, toOnScreen = Camera:WorldToViewportPoint(toPart.Position)
                            
                            if fromOnScreen and toOnScreen then
                                skelData.line.From = Vector2.new(fromPos.X, fromPos.Y)
                                skelData.line.To = Vector2.new(toPos.X, toPos.Y)
                                skelData.line.Color = color
                                skelData.line.Visible = true
                            else
                                skelData.line.Visible = false
                            end
                        else
                            skelData.line.Visible = false
                        end
                    end
                else
                    for _, skelData in pairs(esp.SkeletonLines) do
                        skelData.line.Visible = false
                    end
                end
                end
            else
                -- Off screen
                esp.Box.Visible = false
                esp.BoxOutline.Visible = false
                esp.Name.Visible = false
                esp.Distance.Visible = false
                esp.Health.Visible = false
                esp.Tracer.Visible = false
                for _, skelData in pairs(esp.SkeletonLines) do
                    skelData.line.Visible = false
                end
            end
        end
    end
    
    -- Clean up old ESP
    for character, _ in pairs(ESPObjects) do
        if not activeCharacters[character] then
            RemoveESP(character)
        end
    end
end

RunService.RenderStepped:Connect(UpdateESP)

            pcall(function()
                local char = LocalPlayer.Character
                if char then
                    for _, tool in pairs(char:GetChildren()) do
                        if tool:IsA("Tool") then
                            if _G.NoRecoil then
                                local recoil = tool:FindFirstChild("Recoil") or tool:FindFirstChild("RecoilControl")
                                if recoil and recoil:IsA("NumberValue") then
                                    recoil.Value = 0
                                end
                            end
                            
                            if _G.NoSpread then
                                local spread = tool:FindFirstChild("Spread") or tool:FindFirstChild("Accuracy")
                                if spread and spread:IsA("NumberValue") then
                                    spread.Value = 0
                                end
                            end
                            
                            if _G.InfiniteAmmo then
                                local ammo = tool:FindFirstChild("Ammo") or tool:FindFirstChild("CurrentAmmo")
                                if ammo and ammo:IsA("NumberValue") then
                                    ammo.Value = 999
                                end
                            end
                        end
                    end
                end
            end)
        end
    end
end)

-- TRIGGERBOT
---------------------------------------------------------------------
local TriggerCooldown = false

task.spawn(function()
    RunService.RenderStepped:Connect(function()
        if _G.TriggerBot and not TriggerCooldown then
            -- Check what's under crosshair
            local screenCenter = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)
            local ray = Camera:ViewportPointToRay(screenCenter.X, screenCenter.Y)
            
            local raycastParams = RaycastParams.new()
            raycastParams.FilterType = Enum.RaycastFilterType.Blacklist
            raycastParams.FilterDescendantsInstances = {LocalPlayer.Character, Camera}
            
            local result = workspace:Raycast(ray.Origin, ray.Direction * 500, raycastParams)
            
            if result and result.Instance then
                local hitPart = result.Instance
                local hitModel = hitPart.Parent
                
                if hitModel and hitModel:IsA("Model") then
                    -- Check if it's an enemy player
                    local isEnemy = false
                    
                    -- Check via Players service
                    local targetPlayer = Players:GetPlayerFromCharacter(hitModel)
                    if not targetPlayer then
                        targetPlayer = Players:FindFirstChild(hitModel.Name)
                    end
                    
                    if targetPlayer and targetPlayer ~= LocalPlayer then
                        if _G.TeamCheck then
                            if targetPlayer.Team ~= LocalPlayer.Team then
                                isEnemy = true
                            end
                        else
                            isEnemy = true
                        end
                    else
                        -- Check if it's a camera child (BloXtrike method)
                        for _, model in pairs(Camera:GetChildren()) do
                            if model == hitModel and model.Name ~= LocalPlayer.Name then
                                if not IsOnSameTeam(model) then
                                    isEnemy = true
                                    break
                                end
                            end
                        end
                    end
                    
                    -- Auto shoot if enemy detected
                    if isEnemy then
                        TriggerCooldown = true
                        if mouse1click then
                            mouse1click()
                        end
                        task.wait(_G.TriggerDelay)
                        TriggerCooldown = false
                    end
                end
            end
        end
    end)
end)
