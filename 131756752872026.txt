local Tab = Window:CreateTab("Farm", 4483362458) 



local player = game.Players.LocalPlayer
local char = player.Character or player.CharacterAdded:Wait()
local rootPart = char:WaitForChild("HumanoidRootPart")


local Toggle = Tab:CreateToggle({
   Name = "Auto highest value fish",
   CurrentValue = false,
   Flag = "AutoBestFish",
   Callback = function(Value)
      getgenv().Best = Value

      if Value then
         task.spawn(function()
            while getgenv().Best do
               pcall(function() 

                  local player = game.Players.LocalPlayer
                  local char = player.Character
                  if not char then return end

                  local rootPart = char:FindFirstChild("HumanoidRootPart")
                  if not rootPart then return end

                  local oldCF = rootPart.CFrame

                  local highV = -math.huge
                  local highF = nil

                  for _, v in pairs(workspace.Game.Fishes:GetChildren()) do
                     if v:IsA("Model") then
                        local cps = v:GetAttribute("CashPerSec")
                        if cps and cps > highV then
                           highV = cps
                           highF = v
                        end
                     end
                  end

                  if highF then
                     local fishRoot = highF:FindFirstChild("RootPart", true)
                     if fishRoot then
                        local prompt = fishRoot:FindFirstChildWhichIsA("ProximityPrompt", true)
                        if prompt then
                           rootPart.CFrame = fishRoot.CFrame
                           task.wait(0.3)
                           fireproximityprompt(prompt)
                           task.wait(0.4)
                           rootPart.CFrame = oldCF
                        end
                     end
                  end

               end)

               task.wait(0.2)
            end
         end)
      end
   end,
})






local Button = Tab:CreateButton({

local Section = Tab:CreateSection("Shop")


local Toggle = Tab:CreateToggle({
    Name = "Auto buy Worm",
    CurrentValue = false,
    Flag = "Worm",
    Callback = function(Value)
        getgenv().AutoBuyWorm = Value

        task.spawn(function()
            local player = game.Players.LocalPlayer
            local worm = player.PlayerGui.PersistentUI.Shops.Treat.Content.ScrollingFrame.Worm
            local remote = game:GetService("ReplicatedStorage").Packets.Packet.RemoteEvent

            while true do
                if getgenv().AutoBuyWorm then
                    local stock = worm:GetAttribute("Stock") or 0
                    while stock > 0 do
                        remote:FireServer(buffer.fromstring("\004\005Treat\004Worm"))
                        task.wait(0.05)
                        stock = worm:GetAttribute("Stock") or 0
                    end
                end
                task.wait(0.1)
            end
        end)
    end,
})
