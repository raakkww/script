local function createRainbowBorder()

    local borderThickness = 5

    local borderParts = {}

    

    -- Создаем 4 части обводки (верх, право, низ, лево)

    for i = 1, 4 do

        local part = Instance.new("Frame")

        part.Name = "BorderPart_"..i

        part.BorderSizePixel = 0

        part.ZIndex = 0

        part.Parent = MainContainer

        

        table.insert(borderParts, part)

    end

    

    -- Обновляем цвета обводки

    coroutine.wrap(function()

        while true do

            for hue = 0, 1, 0.01 do

                local color = Color3.fromHSV(hue, 1, 1)

                

                -- Верхняя часть

                borderParts[1].Size = UDim2.new(1, 0, 0, borderThickness)

                borderParts[1].Position = UDim2.new(0, 0, 0, 0)

                borderParts[1].BackgroundColor3 = color

                

                -- Правая часть

                borderParts[2].Size = UDim2.new(0, borderThickness, 1, -borderThickness*2)

                borderParts[2].Position = UDim2.new(1, -borderThickness, 0, borderThickness)

                borderParts[2].BackgroundColor3 = color

                

                -- Нижняя часть

                borderParts[3].Size = UDim2.new(1, 0, 0, borderThickness)

                borderParts[3].Position = UDim2.new(0, 0, 1, -borderThickness)

                borderParts[3].BackgroundColor3 = color

                

                -- Левая часть

                borderParts[4].Size = UDim2.new(0, borderThickness, 1, -borderThickness*2)

                borderParts[4].Position = UDim2.new(0, 0, 0, borderThickness)

                borderParts[4].BackgroundColor3 = color

                

                wait(0.05)

            end

        end

    end)()

end

createRainbowBorder()

-- Функционал перемещения окна

local UserInputService = game:GetService("UserInputService")

local dragging

local dragInput

local dragStart

local startPos

local function update(input)

    local delta = input.Position - dragStart

    MainContainer.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)

end

MainFrame.InputBegan:Connect(function(input)

    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then

        dragging = true

        dragStart = input.Position

        startPos = MainContainer.Position

        

        input.Changed:Connect(function()

            if input.UserInputState == Enum.UserInputState.End then

                dragging = false

            end

        end)

    end

end)

MainFrame.InputChanged:Connect(function(input)

    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then

        dragInput = input

    end

end)

UserInputService.InputChanged:Connect(function(input)

    if input == dragInput and dragging then

        update(input)

    end

end)
