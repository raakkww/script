local FLY_SPEED = 150
local UNDERGROUND_DEPTH = 25
local ARRIVAL_DISTANCE = 5
local PROXIMITY_RANGE = 30

local platformCache = {}

local function createPlatform(position)
    local key = tostring(position)
    if platformCache[key] and platformCache[key].Parent then
        return platformCache[key]
    end
    
    local platform = Instance.new("Part")
    platform.Size = Vector3.new(10, 1, 10)
    platform.Position = position - Vector3.new(0, 3, 0)
    platform.Anchored = true
    platform.CanCollide = true
    platform.BrickColor = BrickColor.new("Bright blue")
    platform.Material = Enum.Material.Neon
    platform.Parent = workspace
    
    platformCache[key] = platform
    return platform
end

local function removeOldPlatforms()
    for _, platform in pairs(platformParts) do
        pcall(function() platform:Destroy() end)
    end
    platformParts = {}
    platformCache = {}
end

local function enableNoclip()
    if noclipConnection then noclipConnection:Disconnect() end
    noclipConnection = RunService.Stepped:Connect(function()
        if teleportEnabled and character then
            for _, part in pairs(character:GetDescendants()) do
                if part:IsA("BasePart") then
                    part.CanCollide = false
                end
            end
        end
    end)
end

local function disableNoclip()
    if noclipConnection then noclipConnection:Disconnect() noclipConnection = nil end
    if character then
        for _, part in pairs(character:GetDescendants()) do
            if part:IsA("BasePart") then
                part.CanCollide = true
            end
        end
    end
end

local function enableProximityFire()
    if proximityConnection then proximityConnection:Disconnect() end
    proximityConnection = RunService.Heartbeat:Connect(function()
        if not teleportEnabled or not character or not humanoidRootPart or not atLocation or not currentLocationPos then 
            return 
        end
        
        local distanceFromLocation = (humanoidRootPart.Position - currentLocationPos).Magnitude
        if distanceFromLocation > ARRIVAL_DISTANCE + 5 then
            return
        end
        
        local playerPos = humanoidRootPart.Position
        for _, v in pairs(workspace:GetDescendants()) do
            if v:IsA("ProximityPrompt") then
                local promptPart = v.Parent
                local partPos = promptPart and (promptPart:IsA("BasePart") and promptPart.Position or (promptPart:IsA("Model") and promptPart.PrimaryPart and promptPart.PrimaryPart.Position))
                if partPos and (playerPos - partPos).Magnitude <= PROXIMITY_RANGE then
                    pcall(function() fireproximityprompt(v) end)
                end
            end
        end
    end)
end

local function disableProximityFire()
    if proximityConnection then proximityConnection:Disconnect() proximityConnection = nil end
end

local function stopBodyVelocity()
    if bodyVelocity and bodyVelocity.Parent then
        bodyVelocity:Destroy()
        bodyVelocity = nil
    end
end

local function flyToLocation(targetPos, callback)
    if not humanoidRootPart then return end
    
    stopBodyVelocity()
    atLocation = false
    currentLocationPos = nil
    humanoid.PlatformStand = true
    
    local currentPos = humanoidRootPart.Position
    local undergroundStart = Vector3.new(currentPos.X, currentPos.Y - UNDERGROUND_DEPTH, currentPos.Z)
    humanoidRootPart.CFrame = CFrame.new(undergroundStart)
    
    local undergroundTarget = Vector3.new(targetPos.X, targetPos.Y - UNDERGROUND_DEPTH, targetPos.Z)
    
    bodyVelocity = Instance.new("BodyVelocity")
    bodyVelocity.MaxForce = Vector3.new(1e9, 1e9, 1e9)
    bodyVelocity.P = 1250
    bodyVelocity.Velocity = (undergroundTarget - undergroundStart).Unit * FLY_SPEED
    bodyVelocity.Parent = humanoidRootPart
    
    local arrival
    arrival = RunService.Heartbeat:Connect(function()
        if not teleportEnabled or not bodyVelocity then
            arrival:Disconnect()
            return
        end
        if (humanoidRootPart.Position - undergroundTarget).Magnitude <= ARRIVAL_DISTANCE then
            stopBodyVelocity()
            humanoidRootPart.CFrame = CFrame.new(targetPos)
            humanoidRootPart.Velocity = Vector3.new(0,0,0)
            humanoid.PlatformStand = false
            atLocation = true
            currentLocationPos = targetPos
            arrival:Disconnect()
            if callback then callback() end
        end
    end)
end

local function updateUI()
    if teleportEnabled then
        statusLabel.Text = "🔵 ON"
        statusLabel.TextColor3 = Color3.fromRGB(100, 255, 100)
        holdButton.Text = "⏹️ HOLD TO STOP"
        holdButton.BackgroundColor3 = Color3.fromRGB(220, 70, 70)
        featuresLabel.TextColor3 = Color3.fromRGB(100, 255, 100)
    else
        statusLabel.Text = "⚪ OFF"
        statusLabel.TextColor3 = Color3.fromRGB(255, 100, 100)
        holdButton.Text = "⚡ HOLD TO START"
        holdButton.BackgroundColor3 = Color3.fromRGB(70, 180, 70)
        timerLabel.Text = ""
        featuresLabel.TextColor3 = Color3.fromRGB(150, 150, 150)
        stopBodyVelocity()
        removeOldPlatforms()
    end
    holdProgress.Size = UDim2.new(0, 0, 1, 0)
end

local function getCurrentLocations()
    if currentMode == "Normal" then return teleportLocations
    elseif currentMode == "Box Farm" then return boxFarmLocations
    else return jewelryFarmLocations end
end

local function waitAtLocation()
    if not teleportEnabled then return end
    for i = 10,1,-1 do
        timerLabel.Text = "⏳ "..i.."s"
        task.wait(1)
        if not teleportEnabled or not atLocation then 
            timerLabel.Text = ""
            return 
        end
    end
    timerLabel.Text = ""
    if teleportEnabled and atLocation then
        currentIndex = currentIndex + 1
        local locations = getCurrentLocations()
        if currentIndex > #locations then currentIndex = 1 end
        locationLabel.Text = "📍 "..currentIndex.."/"..#locations
        
        local nextLocation = locations[currentIndex]
        if currentMode == "Normal" or currentMode == "Jewelry" then
            removeOldPlatforms()
            table.insert(platformParts, createPlatform(nextLocation))
        end
        
        flyToLocation(nextLocation, waitAtLocation)
    end
end

local function startTeleport()
    teleportEnabled = true
    updateUI()
    enableNoclip()
    enableProximityFire()
    
    currentIndex = 1
    local locations = getCurrentLocations()
    locationLabel.Text = "📍 1/"..#locations
    
    local firstLocation = locations[currentIndex]
    if currentMode == "Normal" or currentMode == "Jewelry" then
        table.insert(platformParts, createPlatform(firstLocation))
    end
    
    flyToLocation(firstLocation, waitAtLocation)
end

local function stopTeleport()
    teleportEnabled = false
    updateUI()
    disableNoclip()
    disableProximityFire()
    atLocation = false
    currentLocationPos = nil
    removeOldPlatforms()
    stopBodyVelocity()
    if humanoid then humanoid.PlatformStand = false end
end

local holdActive = false
local holdStartTime = 0
local holdConnection = nil

local function updateHold()
    if holdActive and not teleportEnabled then
        local progress = math.min((tick() - holdStartTime)/1,1)
        holdProgress.Size = UDim2.new(progress,0,1,0)
        if progress >= 1 then 
            holdActive = false 
            if holdConnection then holdConnection:Disconnect() holdConnection = nil end
            startTeleport() 
        end
    elseif holdActive and teleportEnabled then
        local progress = math.min((tick() - holdStartTime)/1,1)
        holdProgress.Size = UDim2.new(progress,0,1,0)
        if progress >= 1 then 
            holdActive = false 
            if holdConnection then holdConnection:Disconnect() holdConnection = nil end
            stopTeleport() 
        end
    end
end

holdButton.MouseButton1Down:Connect(function()
    holdActive = true
    holdStartTime = tick()
    holdConnection = RunService.Heartbeat:Connect(updateHold)
end)

holdButton.MouseButton1Up:Connect(function()
    holdActive = false
    if holdConnection then holdConnection:Disconnect() holdConnection = nil end
    holdProgress.Size = UDim2.new(0,0,1,0)
end)

holdButton.TouchLongPress:Connect(function()
    if not teleportEnabled then startTeleport() else stopTeleport() end
end)

end)
